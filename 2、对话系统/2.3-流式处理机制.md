# 2.3 流式处理机制与实时输出

## 流式处理概述

Claude Code 采用流式处理架构，实现实时输出显示，让用户能够即时看到 API 响应和工具执行结果，而无需等待完整响应。

## 1. 流式处理架构

**文件**: `src/services/streaming/StreamHandler.ts:1-300`

```typescript
/**
 * 流式处理器 - 管理来自 API 的流式响应
 */
export class StreamHandler {
  private decoder: TextDecoder
  private buffer: string = ''

  constructor() {
    this.decoder = new TextDecoder()
  }

  /**
   * 处理流式 API 响应
   */
  async *handleStream(
    response: Response,
    context: StreamContext
  ): AsyncGenerator<StreamEvent, void, unknown> {
    const reader = response.body?.getReader()

    if (!reader) {
      throw new Error('Response body is not readable')
    }

    try {
      // === 阶段 1: 初始化 ===
      yield {
        type: 'start',
        timestamp: Date.now()
      }

      // === 阶段 2: 读取流数据 ===
      while (true) {
        const { done, value } = await reader.read()

        if (done) break

        // === 解码数据块 ===
        const chunk = this.decoder.decode(value, { stream: true })
        this.buffer += chunk

        // === 解析 SSE 事件 ===
        const events = this.parseSSE(this.buffer)

        for (const event of events) {
          yield event

          // 检查上下文中止信号
          if (context.abortController?.signal.aborted) {
            reader.cancel()
            yield { type: 'abort', timestamp: Date.now() }
            return
          }
        }
      }

      // === 阶段 3: 完成 ===
      const finalChunk = this.decoder.decode()
      if (finalChunk) {
        this.buffer += finalChunk
        const finalEvents = this.parseSSE(this.buffer)
        for (const event of finalEvents) {
          yield event
        }
      }

      yield {
        type: 'end',
        timestamp: Date.now()
      }

    } finally {
      reader.releaseLock()
    }
  }

  /**
   * 解析 Server-Sent Events (SSE) 格式
   *
   * 格式：
   * event: content_block_start
   * data: {...json...}
   *
   * event: content_block_delta
   * data: {...json...}
   */
  private parseSSE(data: string): StreamEvent[] {
    const events: StreamEvent[] = []
    const lines = data.split('\n')

    let currentEvent: {
      type?: string
      data?: string
    } = {}

    for (const line of lines) {
      if (line.startsWith('event: ')) {
        if (currentEvent.type && currentEvent.data) {
          events.push(this.buildEvent(currentEvent))
        }
        currentEvent.type = line.slice(7).trim()
      } else if (line.startsWith('data: ')) {
        const jsonStr = line.slice(6).trim()
        try {
          currentEvent.data = jsonStr
        } catch (error) {
          console.error('Failed to parse SSE data:', error)
        }
      } else if (line === '') {
        // 空行表示事件结束
        if (currentEvent.type && currentEvent.data) {
          events.push(this.buildEvent(currentEvent))
          currentEvent = {}
        }
      }
    }

    // 清空缓冲区中已解析的部分
    const lastIndex = data.lastIndexOf('\n\n')
    if (lastIndex >= 0) {
      this.buffer = data.slice(lastIndex + 2)
    }

    return events
  }

  /**
   * 构建流事件对象
   */
  private buildEvent(parsed: {
    type?: string
    data?: string
  }): StreamEvent {
    if (!parsed.type || !parsed.data) {
      return { type: 'unknown', timestamp: Date.now() }
    }

    try {
      const data = JSON.parse(parsed.data)

      switch (parsed.type) {
        case 'message_start':
          return {
            type: 'message_start',
            messageId: data.message.id,
            timestamp: Date.now()
          }

        case 'content_block_start':
          return {
            type: 'content_block_start',
            blockType: data.content_block.type,
            blockIndex: data.index,
            timestamp: Date.now()
          }

        case 'content_block_delta':
          if (data.delta.type === 'text_delta') {
            return {
              type: 'text_delta',
              text: data.delta.text,
              blockIndex: data.index,
              timestamp: Date.now()
            }
          } else if (data.delta.type === 'input_json_delta') {
            return {
              type: 'json_delta',
              json: data.delta.partial_json,
              blockIndex: data.index,
              timestamp: Date.now()
            }
          }
          break

        case 'content_block_stop':
          return {
            type: 'content_block_stop',
            blockIndex: data.index,
            timestamp: Date.now()
          }

        case 'message_delta':
          return {
            type: 'message_delta',
            deltaType: data.delta.type,
            data: data.delta,
            timestamp: Date.now()
          }

        case 'message_stop':
          return {
            type: 'message_stop',
            timestamp: Date.now()
          }
      }
    } catch (error) {
      console.error('Failed to build event:', error)
    }

    return { type: 'unknown', timestamp: Date.now() }
  }
}

/**
 * 流事件类型
 */
export type StreamEvent =
  | { type: 'start'; timestamp: number }
  | { type: 'message_start'; messageId: string; timestamp: number }
  | { type: 'content_block_start'; blockType: string; blockIndex: number; timestamp: number }
  | { type: 'text_delta'; text: string; blockIndex: number; timestamp: number }
  | { type: 'json_delta'; json: string; blockIndex: number; timestamp: number }
  | { type: 'content_block_stop'; blockIndex: number; timestamp: number }
  | { type: 'message_delta'; deltaType: string; data: unknown; timestamp: number }
  | { type: 'message_stop'; timestamp: number }
  | { type: 'abort'; timestamp: number }
  | { type: 'end'; timestamp: number }
  | { type: 'unknown'; timestamp: number }

/**
 * 流处理上下文
 */
export interface StreamContext {
  abortController?: AbortController
  onProgress?: (event: StreamEvent) => void
  onError?: (error: Error) => void
}
```

```typescript
/**
 * 事件总线 - 分发流事件
 */
export class EventBus {
  private handlers = new Map<string, Set<Handler>>()
  private eventHistory: StreamEvent[] = []
  private maxHistory = 1000

  /**
   * 订阅事件
   */
  subscribe(eventType: string, handler: Handler): () => void {
    if (!this.handlers.has(eventType)) {
      this.handlers.set(eventType, new Set())
    }

    this.handlers.get(eventType)!.add(handler)

    // 返回取消订阅函数
    return () => {
      this.handlers.get(eventType)?.delete(handler)
    }
  }

  /**
   * 发送事件
   */
  async emit(event: StreamEvent): Promise<void> {
    // 保存到历史
    this.eventHistory.push(event)
    if (this.eventHistory.length > this.maxHistory) {
      this.eventHistory.shift()
    }

    // 分发给订阅者
    const handlers = this.handlers.get(event.type)
    if (handlers) {
      await Promise.all(
        Array.from(handlers).map(handler =>
          handler(event).catch(err => {
            console.error(`Error in handler for ${event.type}:`, err)
          })
        )
      )
    }
  }

  /**
   * 获取事件历史
   */
  getHistory(type?: string): StreamEvent[] {
    if (type) {
      return this.eventHistory.filter(e => e.type === type)
    }
    return [...this.eventHistory]
  }

  /**
   * 清空历史
   */
  clearHistory(): void {
    this.eventHistory = []
  }
}

type Handler = (event: StreamEvent) => Promise<void> | void
```

```typescript
/**
 * 流式输出显示组件（React/Ink）
 */
export interface StreamDisplayProps {
  messageId: string
  onComplete?: () => void
}

export const StreamDisplay: React.FC<StreamDisplayProps> = ({
  messageId,
  onComplete
}) => {
  // === 状态管理 ===
  const [content, setContent] = React.useState<string>('')
  const [toolCalls, setToolCalls] = React.useState<ToolUseBlock[]>([])
  const [isComplete, setIsComplete] = React.useState(false)
  const [blockIndex, setBlockIndex] = React.useState(0)

  // === 流事件处理 ===
  React.useEffect(() => {
    const eventBus = getEventBus()

    // 监听文本增量
    const unsubText = eventBus.subscribe('text_delta', (event: any) => {
      setContent(prev => prev + event.text)
    })

    // 监听 JSON（工具调用）
    const unsubJson = eventBus.subscribe('json_delta', (event: any) => {
      // 累积 JSON 片段，直到完成一个完整的工具调用
      setToolCalls(prev => {
        const updated = [...prev]
        if (!updated[event.blockIndex]) {
          updated[event.blockIndex] = {
            type: 'tool_use',
            id: '',
            name: '',
            input: {}
          }
        }
        // 增量更新 JSON
        updated[event.blockIndex] = {
          ...updated[event.blockIndex],
          ...JSON.parse(event.json)
        }
        return updated
      })
    })

    // 监听块完成
    const unsubStop = eventBus.subscribe(
      'content_block_stop',
      (event: any) => {
        setBlockIndex(event.blockIndex + 1)
      }
    )

    // 监听消息完成
    const unsubEnd = eventBus.subscribe('message_stop', () => {
      setIsComplete(true)
      onComplete?.()
    })

    return () => {
      unsubText()
      unsubJson()
      unsubStop()
      unsubEnd()
    }
  }, [onComplete])

  // === 渲染逻辑 ===
  return (
    <Box flexDirection="column">
      {/* 文本内容 */}
      {content && (
        <Box>
          <Text>{content}</Text>
        </Box>
      )}

      {/* 工具调用 */}
      {toolCalls.map((tool, idx) => (
        <Box key={idx} marginLeft={2}>
          <Text color="yellow">
            🔧 {tool.name}
          </Text>
          {isComplete && (
            <Text color="green"> ✓</Text>
          )}
        </Box>
      ))}

      {/* 加载指示符 */}
      {!isComplete && (
        <Text color="gray">⏳ Waiting for response...</Text>
      )}
    </Box>
  )
}
```

```typescript
/**
 * 并行工具执行管理器
 */
export class ParallelToolExecutor {
  private concurrencyLimit: number
  private executingTools = new Map<string, Promise<any>>()

  constructor(limit: number = 20) {
    this.concurrencyLimit = limit
  }

  /**
   * 执行多个工具调用（并行）
   */
  async executeTools(
    toolCalls: ToolUseBlock[],
    tools: Tool[],
    context: ToolUseContext
  ): Promise<ToolResult[]> {
    // === 创建执行队列 ===
    const queue: Array<{
      tool: ToolUseBlock
      execute: () => Promise<ToolResult>
    }> = []

    for (const toolCall of toolCalls) {
      const tool = tools.find(t => t.name === toolCall.name)
      if (!tool) {
        queue.push({
          tool: toolCall,
          execute: async () => ({
            type: 'error' as const,
            error: `Tool not found: ${toolCall.name}`
          })
        })
        continue
      }

      queue.push({
        tool: toolCall,
        execute: () => this.executeWithTimeout(tool, toolCall.input, context)
      })
    }

    // === 使用 pLimit 管理并发 ===
    const limit = pLimit(this.concurrencyLimit)

    const results = await Promise.all(
      queue.map(item =>
        limit(() => this.executeAndReport(item.tool, item.execute, context))
      )
    )

    return results
  }

  /**
   * 执行单个工具（带超时）
   */
  private async executeWithTimeout(
    tool: Tool,
    input: Record<string, unknown>,
    context: ToolUseContext
  ): Promise<ToolResult> {
    const timeout = tool.timeout ?? 30000

    return Promise.race([
      // 正常执行
      tool.invoke(input, context),

      // 超时保护
      new Promise<ToolResult>((_, reject) =>
        setTimeout(
          () => reject(new Error(`Tool timeout: ${tool.name}`)),
          timeout
        )
      )
    ]).catch(error => ({
      type: 'error' as const,
      error: String(error),
      errorCode: 'timeout'
    }))
  }

  /**
   * 执行并报告进度
   */
  private async executeAndReport(
    toolCall: ToolUseBlock,
    execute: () => Promise<ToolResult>,
    context: ToolUseContext
  ): Promise<ToolResult> {
    const startTime = Date.now()

    try {
      const result = await execute()

      const metrics: ToolMetrics = {
        toolName: toolCall.name,
        executionTime: Date.now() - startTime,
        success: result.type !== 'error'
      }

      context.reportMetrics?.(metrics)

      return result
    } catch (error) {
      return {
        type: 'error' as const,
        error: String(error),
        errorCode: 'execution_error'
      }
    }
  }
}

/**
 * 导入 pLimit（外部库）
 */
import pLimit from 'p-limit'
```

### 6.1 缓冲区管理

```typescript
/**
 * 流缓冲区优化
 */
const BUFFER_CONFIG = {
  // 最大缓冲大小
  MAX_BUFFER_SIZE: 64 * 1024,  // 64KB

  // 超过此大小后自动冲刷
  FLUSH_THRESHOLD: 32 * 1024,  // 32KB

  // 最长等待时间
  MAX_WAIT_MS: 100,

  // 最小批处理大小
  MIN_BATCH_SIZE: 1024  // 1KB
}

/**
 * 自适应缓冲
 */
export class AdaptiveBuffer {
  private buffer = ''
  private flushTimer: NodeJS.Timeout | null = null

  async write(chunk: string): Promise<void> {
    this.buffer += chunk

    // 缓冲区过大，立即冲刷
    if (this.buffer.length > BUFFER_CONFIG.FLUSH_THRESHOLD) {
      await this.flush()
    } else if (!this.flushTimer) {
      // 设置定时器，等待更多数据
      this.flushTimer = setTimeout(() => {
        this.flush()
        this.flushTimer = null
      }, BUFFER_CONFIG.MAX_WAIT_MS)
    }
  }

  async flush(): Promise<void> {
    if (this.flushTimer) {
      clearTimeout(this.flushTimer)
      this.flushTimer = null
    }

    if (this.buffer.length === 0) return

    // 处理缓冲区内容
    const content = this.buffer
    this.buffer = ''

    // 触发事件
    await getEventBus().emit({
      type: 'text_delta',
      text: content,
      blockIndex: 0,
      timestamp: Date.now()
    })
  }
}
```

✅ **实时显示** - 流事件即时渲染，用户实时看到响应
✅ **SSE 解析** - 标准 Server-Sent Events 格式处理
