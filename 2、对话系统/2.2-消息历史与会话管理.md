export async function createSession(
  options: CreateSessionOptions = {}

  const {
    initialMessages = [],
    model = 'claude-3-5-sonnet-20241022',
    maxTokens = 100_000,
    customSystemPrompt,
    enableMemory = true,
    enableAutoSave = true,

  } = options

  // === 步骤 1: 生成会话 ID ===

  // === 步骤 2: 初始化元数据 ===
  const metadata: SessionMetadata = {
    id: sessionId,
    createdAt: Date.now(),
    updatedAt: Date.now(),
    type: sessionType,
    model,
    maxTokens,
    status: 'active',
    statistics: {
      messageCount: initialMessages.length,
      totalTokens: 0,
      totalCost: 0
    }
  }

  // === 步骤 3: 加载或初始化消息历史 ===
  const messages: Message[] = [
    // 系统消息
    {
      type: 'system',
      uuid: generateUUID(),
      content: 'Session initialized',
      category: 'info',
      timestamp: Date.now()
    },
    // 初始消息（如果有）
    ...initialMessages
  ]

  // === 步骤 4: 初始化应用状态 ===
  const appState: AppState = {
    sessionId,
    isRemoteSession: options.isRemote || false,
    messages,
    isWaitingForResponse: false,
    tools: await loadTools(),
    skills: await loadSkills(),
    settings: await loadUserSettings(),
    // ... 其他初始状态
  }

  // === 步骤 5: 创建会话对象 ===
  const session: Session = {
    id: sessionId,
    metadata,
    appState,
    history: {
      sessionId,
      messages,
      metadata: {
        createdAt: metadata.createdAt,
        lastModified: metadata.updatedAt,
        totalTokens: 0,
        model
      }
    },

    // === 会话方法 ===

    /**
     * 添加消息到会话
     */
    async addMessage(message: Message) {
      this.history.messages.push(message)
      this.metadata.statistics.messageCount++
      this.metadata.updatedAt = Date.now()

      // 自动保存
      if (enableAutoSave) {
        await this.save()
      }
    },

    /**
     * 获取消息历史
     */
    getMessages(): Message[] {
      return [...this.history.messages]
    },

    /**
     * 压缩消息历史
     */
    async compress() {
      const compressed = compressMessages(
        this.history.messages,
        0.7 // 保留 70% 的 token
      )
      this.history.messages = compressed
      await this.save()
    },

    /**
     * 保存会话到磁盘
     */
    async save() {
      const sessionPath = getSessionPath(this.id)
      const data = {
        metadata: this.metadata,
        history: this.history
      }

      await writeFile(
        sessionPath,
        JSON.stringify(data, null, 2)
      )
    },

    /**
     * 导出会话
     */
    async export(format: 'json' | 'markdown' = 'json') {
      if (format === 'json') {
        return JSON.stringify(this.history, null, 2)
      } else {
        // 导出为 Markdown
        return messagestoMarkdown(this.history.messages)
      }
    },

    /**
     * 清除会话
     */
    async clear() {
      this.history.messages = []
      this.metadata.statistics.messageCount = 0
      this.metadata.statistics.totalTokens = 0
      await this.save()
    }
  }

  // === 步骤 6: 保存会话 ===
  await session.save()

  return session
}

// src/history.ts:250-350
/**
 * JSONL 格式的增量写入策略
 * 每条消息都是一行独立的 JSON，便于流式读取
 */

export class MessageHistoryPersistence {
  private sessionId: string
  private historyFile: string
  private writeStream?: WriteStream
  private buffer: string[] = []
  private bufferSize = 0
  private flushThreshold = 64 * 1024 // 64KB

  constructor(sessionId: string) {
    this.sessionId = sessionId
    this.historyFile = SESSION_STORAGE_PATHS.historyFile(sessionId)
  }

  /**
   * 初始化写入
   */
  async init() {
    await ensureDir(path.dirname(this.historyFile))
  }

  /**
   * 追加消息到历史
   */
  async appendMessage(message: Message) {
    // 将消息序列化为 JSON
    const line = JSON.stringify(message)

    // 缓冲写入
    this.buffer.push(line)
    this.bufferSize += line.length

    // 当缓冲区达到阈值时刷新
    if (this.bufferSize >= this.flushThreshold) {
      await this.flush()
    }
  }

  /**
   * 批量追加消息
   */
  async appendMessages(messages: Message[]) {
    for (const message of messages) {
      await this.appendMessage(message)
    }
  }

  /**
   * 刷新缓冲区到磁盘
   */
  async flush() {
    if (this.buffer.length === 0) {
      return
    }

    const content = this.buffer.join('\n') + '\n'
    await appendFile(this.historyFile, content)

    this.buffer = []
    this.bufferSize = 0
  }

  /**
   * 关闭持久化存储
   */
  async close() {
    await this.flush()
  }

  /**
   * 加载消息历史
   */
  async loadMessages(
    limit?: number,
    offset?: number
  ): Promise<Message[]> {
    try {
      const content = await readFile(this.historyFile, 'utf-8')
      const lines = content.split('\n').filter(l => l.trim())

      // 解析 JSONL
      let messages = lines.map(line => JSON.parse(line) as Message)

      // 应用偏移和限制
      if (offset) {
        messages = messages.slice(offset)
      }
      if (limit) {
        messages = messages.slice(0, limit)
      }

      return messages
    } catch (error) {
      if ((error as any).code === 'ENOENT') {
        return [] // 文件不存在
      }
      throw error
    }
  }

  /**
   * 获取消息总数（快速）
   */
  async getMessageCount(): Promise<number> {
    try {
      const content = await readFile(this.historyFile, 'utf-8')
      return content.split('\n').filter(l => l.trim()).length
    } catch {
      return 0
    }
  }

  /**
   * 流式读取消息
   */
  async* streamMessages(
    batchSize: number = 100
  ): AsyncGenerator<Message[]> {
    const content = await readFile(this.historyFile, 'utf-8')
    const lines = content.split('\n').filter(l => l.trim())

    for (let i = 0; i < lines.length; i += batchSize) {
      const batch = lines
        .slice(i, i + batchSize)
        .map(line => JSON.parse(line) as Message)

      yield batch
    }
  }
}

/**
 * 定期保存检查点，便于快速恢复
 */

export interface Checkpoint {
  id: string
  timestamp: number
  messageCount: number
  lastMessageUuid: string
  totalTokens: number
  compressedSize: number
}

export class CheckpointManager {
  private sessionId: string
  private checkpointsDir: string

  constructor(sessionId: string) {
    this.sessionId = sessionId
    this.checkpointsDir = SESSION_STORAGE_PATHS.checkpointsDir(sessionId)
  }

  /**
   * 创建检查点
   */
  async createCheckpoint(
    messages: Message[],
    totalTokens: number
  ): Promise<Checkpoint> {
    const checkpointId = generateCheckpointId()

    // 压缩消息
    const compressed = Buffer.from(
      JSON.stringify(messages)
    )

    // 保存到检查点文件
    const checkpointPath = join(
      this.checkpointsDir,
      `${checkpointId}.json.gz`
    )

    await ensureDir(this.checkpointsDir)

    // 使用 gzip 压缩
    const compressed$ = createGzip()
    const writeStream = createWriteStream(checkpointPath)

    await new Promise((resolve, reject) => {
      compressed$.pipe(writeStream)
        .on('finish', resolve)
        .on('error', reject)

      compressed$.write(compressed)
      compressed$.end()
    })

    const checkpoint: Checkpoint = {
      id: checkpointId,
      timestamp: Date.now(),
      messageCount: messages.length,
      lastMessageUuid: messages[messages.length - 1]?.uuid || '',
      totalTokens,
      compressedSize: (await stat(checkpointPath)).size
    }

    // 保存元数据
    await this.saveCheckpointMetadata(checkpoint)

    return checkpoint
  }

  /**
   * 从检查点恢复
   */
  async restoreFromCheckpoint(
    checkpointId: string
  ): Promise<Message[]> {
    const checkpointPath = join(
      this.checkpointsDir,
      `${checkpointId}.json.gz`
    )

    // 解压并读取
    const chunks: Buffer[] = []

    await new Promise((resolve, reject) => {
      createReadStream(checkpointPath)
        .pipe(createGunzip())
        .on('data', chunk => chunks.push(chunk))
        .on('end', resolve)
        .on('error', reject)
    })

    const buffer = Buffer.concat(chunks)
    const messages = JSON.parse(buffer.toString()) as Message[]

    return messages
  }

  /**
   * 列出所有检查点
   */
  async listCheckpoints(): Promise<Checkpoint[]> {
    try {
      const metadataPath = join(this.checkpointsDir, 'metadata.json')
      const content = await readFile(metadataPath, 'utf-8')
      return JSON.parse(content) as Checkpoint[]
    } catch {
      return []
    }
  }

  /**
   * 保存检查点元数据
   */
  private async saveCheckpointMetadata(checkpoint: Checkpoint) {
    const metadataPath = join(this.checkpointsDir, 'metadata.json')

    let metadata: Checkpoint[] = []
    try {
      const content = await readFile(metadataPath, 'utf-8')
      metadata = JSON.parse(content)
    } catch {
      // 文件不存在，创建新的
    }

    metadata.push(checkpoint)

    // 保持最近 10 个检查点
    if (metadata.length > 10) {
      metadata = metadata.slice(-10)
    }

    await writeFile(
      metadataPath,
      JSON.stringify(metadata, null, 2)
    )
  }
}

export async function loadSession(
  sessionId: string

  try {
    // === 步骤 1: 验证会话存在 ===
    const sessionDir = SESSION_STORAGE_PATHS.sessionDir(sessionId)
    if (!await pathExists(sessionDir)) {
      return null

    const metadataContent = await readFile(metadataPath, 'utf-8')

    // === 步骤 3: 加载消息历史 ===
    const persistence = new MessageHistoryPersistence(sessionId)
    const messages = await persistence.loadMessages()

    // === 步骤 4: 重建应用状态 ===
    const appState = await rebuildAppState(sessionId, messages)

    // === 步骤 5: 重建会话对象 ===
    const session: Session = {

      metadata,
      appState,
      history: {
        sessionId,
        messages,

          lastModified: metadata.updatedAt,

          model: metadata.model

    // === 步骤 6: 验证完整性 ===
    if (!validateSessionIntegrity(session)) {
      // 记录警告，但仍然返回会话
      console.warn(
        `Session ${sessionId} integrity check failed, ` +
        `but continuing with available data`
      )
    }

    return session
  } catch (error) {
    console.error(`Failed to load session ${sessionId}:`, error)
    return null
  }
}

/**
 * 验证会话完整性
 */
function validateSessionIntegrity(session: Session): boolean {
  const { messages, metadata } = session.history

  // 检查 1: 消息数量匹配
  if (messages.length !== metadata.totalMessages) {
    console.warn(
      `Message count mismatch: ` +
      `${messages.length} messages, ` +
      `but metadata says ${metadata.totalMessages}`
    )
  }

  // 检查 2: 消息 UUID 唯一性
  const uuids = new Set<string>()
  for (const message of messages) {
    if (uuids.has(message.uuid)) {
      console.error(
        `Duplicate UUID found: ${message.uuid}`
      )
      return false
    }
    uuids.add(message.uuid)
  }

  // 检查 3: 工具结果消息的有效性
  const toolUseIds = new Set<string>()
  for (const message of messages) {
    if (message.type === 'assistant') {
      for (const block of message.content) {
        if (block.type === 'tool_use') {
          toolUseIds.add(block.id)
        }
      }
    } else if (message.type === 'tool_result') {
      if (!toolUseIds.has(message.toolUseId)) {
        console.warn(
          `Tool result references unknown tool_use: ${message.toolUseId}`
        )
      }
    }
  }

  // 检查 4: 时间戳单调性
  let lastTimestamp = 0
  for (const message of messages) {
    if (message.timestamp < lastTimestamp) {
      console.warn(
        `Non-monotonic timestamps detected: ` +
        `${lastTimestamp} -> ${message.timestamp}`
      )
    }
    lastTimestamp = message.timestamp
  }

  return true
}

async function rebuildSessionIndex(): Promise<SessionMetadata[]> {
  const sessionsDir = SESSION_STORAGE_PATHS.sessionsDir()
  const entries = await readdir(sessionsDir)

  const metadata: SessionMetadata[] = []

      // 忽略无效的会话目录
    }
  }

  // 保存索引
  const indexFile = SESSION_STORAGE_PATHS.indexFile()
  await writeFile(
    indexFile,
    JSON.stringify(metadata, null, 2)
  )

  return metadata
}

// src/history.ts:800-900
/**
 * 文件锁确保并发安全
 * 使用 proper-lockfile 库
 */

import lockfile from 'proper-lockfile'

export class ConcurrentHistoryManager {
  private lockPath: string
  private lock: { release: () => Promise<void> } | null = null

  constructor(sessionId: string) {
    this.lockPath = SESSION_STORAGE_PATHS.historyFile(sessionId) + '.lock'
  }

  /**
   * 获取锁
   */
  async acquireLock(timeout: number = 5000) {
    try {
      this.lock = await lockfile.lock(this.lockPath, {
        stale: 10000,
        onCompromised: (error) => {
          console.error('Lock compromised:', error)
        }
      })
    } catch (error) {
      throw new Error(`Failed to acquire lock after ${timeout}ms`)
    }
  }

  /**
   * 释放锁
   */
  async releaseLock() {
    if (this.lock) {
      await this.lock.release()
      this.lock = null
    }
  }

  /**
   * 原子操作：读取后修改
   */
  async readModifyWrite(
    modifier: (messages: Message[]) => Message[]
  ): Promise<void> {
    await this.acquireLock()

    try {
      // 读取
      const messages = await this.loadMessages()

      // 修改
      const updated = modifier(messages)

      // 写入
      await this.writeMessages(updated)
    } finally {
      await this.releaseLock()

/**
 * 构建消息索引以加快查询
 */

export class MessageIndex {
  private index: {
    byUuid: Map<string, number>         // UUID → 消息索引
    byToolUseId: Map<string, number[]>  // ToolUseId → 消息索引
    byTimestamp: Map<number, number>    // 时间戳 → 消息索引
    byType: Map<string, number[]>       // 类型 → 消息索引数组
  }

  constructor() {
    this.index = {
      byUuid: new Map(),
      byToolUseId: new Map(),
      byTimestamp: new Map(),
      byType: new Map()
    }
  }

  /**
   * 构建索引
   */
  build(messages: Message[]) {
    this.index.byUuid.clear()
    this.index.byToolUseId.clear()
    this.index.byTimestamp.clear()
    this.index.byType.clear()

    for (let i = 0; i < messages.length; i++) {
      const message = messages[i]

      // 按 UUID 索引
      this.index.byUuid.set(message.uuid, i)

      // 按时间戳索引
      this.index.byTimestamp.set(message.timestamp, i)

      // 按类型索引
      if (!this.index.byType.has(message.type)) {
        this.index.byType.set(message.type, [])
      }
      this.index.byType.get(message.type)!.push(i)

      // 按 ToolUseId 索引（如果适用）
      if (message.type === 'tool_result') {
        const toolUseId = (message as ToolResultMessage).toolUseId
        if (!this.index.byToolUseId.has(toolUseId)) {
          this.index.byToolUseId.set(toolUseId, [])
        }
        this.index.byToolUseId.get(toolUseId)!.push(i)
      }
    }
  }

  /**
   * 按 UUID 查找消息
   */
  findByUuid(uuid: string, messages: Message[]): Message | null {
    const index = this.index.byUuid.get(uuid)
    return index !== undefined ? messages[index] : null
  }

  /**
   * 按类型查找消息
   */
  findByType(type: string, messages: Message[]): Message[] {
    const indices = this.index.byType.get(type) || []
    return indices.map(i => messages[i])
  }

  /**
   * 查找特定工具的所有结果
   */
  findToolResults(toolUseId: string, messages: Message[]): ToolResultMessage[] {
    const indices = this.index.byToolUseId.get(toolUseId) || []
    return indices.map(i => messages[i] as ToolResultMessage)
  }
}

export async function repairCorruptedHistory(
  sessionId: string
): Promise<Message[]> {
  const historyFile = SESSION_STORAGE_PATHS.historyFile(sessionId)

  try {
    const content = await readFile(historyFile, 'utf-8')
    const lines = content.split('\n')

    const validMessages: Message[] = []
    const skippedLines: number[] = []

    for (let i = 0; i < lines.length; i++) {
      const line = lines[i].trim()
      if (!line) continue

      try {
        const message = JSON.parse(line) as Message
        validMessages.push(message)
      } catch (error) {
        skippedLines.push(i)
        console.warn(`Skipped invalid line ${i}: ${line.substring(0, 100)}...`)
      }
    }

    // 如果有跳过的行，重写文件
    if (skippedLines.length > 0) {
      console.log(
        `Rewriting history file: skipped ${skippedLines.length} invalid lines`
      )

      const validLines = validMessages
        .map(m => JSON.stringify(m))
        .join('\n')

      await writeFile(historyFile, validLines + '\n')
    }

    return validMessages
  } catch (error) {
    console.error('Failed to repair history:', error)
    throw error
  }
}

export async function recoverFromBackup(
  sessionId: string
): Promise<Message[]> {
  const checkpointManager = new CheckpointManager(sessionId)
  const checkpoints = await checkpointManager.listCheckpoints()

  if (checkpoints.length === 0) {
    throw new Error('No checkpoints available for recovery')
  }

  // 使用最新的检查点
  const latestCheckpoint = checkpoints[checkpoints.length - 1]
  console.log(`Recovering from checkpoint: ${latestCheckpoint.id}`)

  return await checkpointManager.restoreFromCheckpoint(latestCheckpoint.id)
}
