## 概述

Claude Code 的对话系统是一个**事件驱动的流式处理引擎**，实现了用户输入 → API 调用 → 工具执行 → 结果反馈 的完整循环。核心特点：

- **异步迭代执行**：支持多轮工具调用和模型推理
- **流式响应**：实时显示模型输出，不阻塞 UI
- **工具并行执行**：多个工具调用同时进行
- **断点恢复**：支持中止和继续对话

### 核心指标
- **平均响应延迟**：< 500ms 首字显示
- **工具执行并行度**：最多 20 个并发工具
- **消息历史大小**：可配置，默认 100+ 条消息
- **Token 预算**：动态调整，防止上下文溢出

```typescript
  try {
    const [branch, status, log] = await Promise.all([
      // 获取当前分支
```

```typescript
// src/utils/queryContext.ts:200-280
interface ProjectNode {
  path: string
  isDirectory: boolean
  children?: ProjectNode[]
  size?: number
  relevance?: 'high' | 'medium' | 'low'
}

export function buildProjectStructure(
  rootPath: string,
  options: BuildOptions = {}
): ProjectNode {
  const { maxDepth = 3, ignore = [] } = options

  function traverse(
    dirPath: string,
    currentDepth: number
  ): ProjectNode {
    if (currentDepth >= maxDepth) {
      return { path: dirPath, isDirectory: true }
    }

    const entries = fs.readdirSync(dirPath)
    const children: ProjectNode[] = []

    for (const entry of entries) {
      // 忽略规则
      if (ignore.some(pattern => minimatch(entry, pattern))) {
        continue
      }

      const fullPath = path.join(dirPath, entry)
      const stat = fs.statSync(fullPath)

      if (stat.isDirectory()) {
        children.push(traverse(fullPath, currentDepth + 1))
      } else {
        // 标记相关度高的文件
        const relevance = getFileRelevance(entry)
        children.push({
          path: fullPath,
          isDirectory: false,
          size: stat.size,
          relevance
        })
      }
    }

    return { path: dirPath, isDirectory: true, children }
  }

  return traverse(rootPath, 0)
}

function getFileRelevance(filename: string): 'high' | 'medium' | 'low' {
  // 关键文件标记为高相关度
  const highRelevance = [
    'README.md',
    'package.json',
    'tsconfig.json',
    'src/main.ts',
    'Makefile',
    'docker-compose.yml'
  ]

  if (highRelevance.some(f => filename.endsWith(f))) {
    return 'high'
  }

  if (filename.endsWith('.md')) return 'medium'
  if (filename.startsWith('.')) return 'low'

  return 'medium'
}
```

## 流式处理机制

### 1. 生成器模式 (AsyncGenerator)

**文件**：`src/query.ts:100-250`

```typescript
// src/query.ts:100-150
/**
 * 核心查询执行函数
 * 返回异步生成器，支持流式处理
 */
export async function* queryInternal(
  params: QueryParams
): AsyncGenerator<
  StreamEvent | Message,
  Terminal | Continue,
  undefined
> {
  const {
    messages,
    systemPrompt,
    tools,
    toolUseContext,
    maxTurns = 20
  } = params

  let iteration = 0
  const messageHistory = [...messages]

  // 主循环：最多执行 maxTurns 轮
  while (iteration < maxTurns) {
    iteration++

    // 1. 调用 Claude API（流式）
    const stream = await callClaudeAPI({
      system: systemPrompt,
      messages: normalizeMessagesForAPI(messageHistory),
      tools: tools,
      stream: true,
      maxOutputTokens: 8096
    })

    // 2. 流式处理响应
    let assistantMessage: AssistantMessage = {
      uuid: generateUUID(),
      role: 'assistant',
      content: [],
      stopReason: 'end_turn',
      createdAt: Date.now()
    }

    let hasToolUse = false

    // 关键：逐块处理响应
    for await (const event of stream) {
      // 处理不同类型的事件
      if (event.type === 'content_block_delta') {
        if (event.delta.type === 'text_delta') {
          // 立即 yield 文本，不等待完成
          yield {
            type: 'stream_text',
            event,
            text: event.delta.text
          }

          assistantMessage.content.push({
            type: 'text',
            text: event.delta.text
          })
        }
      }

      // 工具调用块
      if (event.type === 'content_block_stop') {
        if (event.content_block.type === 'tool_use') {
          hasToolUse = true
          assistantMessage.content.push(event.content_block)

          // 报告工具检测
          yield {
            type: 'tool_use_detected',
            toolName: event.content_block.name
          }
        }
      }

      // Token 使用情况
      if (event.type === 'message_delta') {
        assistantMessage.stopReason = event.message.stop_reason

        yield {
          type: 'token_usage',
          inputTokens: event.message.usage?.input_tokens,
          outputTokens: event.message.usage?.output_tokens
        }
      }
    }

    // 3. 添加助手消息到历史
    messageHistory.push(assistantMessage)
    yield { type: 'message', message: assistantMessage }

    // 4. 检查是否需要继续
    if (!hasToolUse || assistantMessage.stopReason === 'end_turn') {
      return {
        type: 'terminal',
        reason: 'model_stopped',
        finalMessage: assistantMessage
      }
    }

    // 5. 执行工具（如果有）
    const toolUses = assistantMessage.content.filter(
      c => c.type === 'tool_use'
    ) as ToolUseBlock[]

    // 关键：并行执行工具，而不是顺序
    const toolResults = await runToolsParallel(
      toolUses,
      toolUseContext,
      iteration
    )

    // 6. 添加工具结果到历史
    messageHistory.push({
      role: 'user',
      content: toolResults
    })

    // 继续循环...
  }

  // 达到最大轮数
  return {
    type: 'terminal',
    reason: 'max_turns_exceeded'
  }
}
```

```typescript
export type RetryableError =
  | 'rate_limit'
  | 'server_error'
  | 'timeout'
  | 'auth_error'
  | 'network_error'

export type NonRetryableError =
  | 'invalid_request'
  | 'invalid_api_key'
  | 'model_not_found'
  | 'permission_denied'

export function categorizeAPIError(error: Error): RetryableError | NonRetryableError | null {
  const message = error.message.toLowerCase()

  // 速率限制
  if (message.includes('rate_limit') || message.includes('429')) {
    return 'rate_limit'
  }

  // 服务器错误
  if (message.includes('500') || message.includes('502') || message.includes('503')) {
    return 'server_error'
  }

  // 超时
  if (message.includes('timeout') || message.includes('econnreset')) {
    return 'timeout'
  }

  // 认证错误
  if (message.includes('401') || message.includes('unauthorized')) {
    return 'auth_error'
  }

  // 网络错误
  if (message.includes('enotfound') || message.includes('econnrefused')) {
    return 'network_error'
  }

  // 无效请求
  if (message.includes('400') || message.includes('invalid')) {
    return 'invalid_request'
  }

  return null
}

export async function queryWithRetry(
  params: QueryParams,
  options: RetryOptions = {}
): Promise<QueryResult> {
  const {
    maxRetries = 3,
    baseDelay = 1000,
    maxDelay = 32000,
    backoffMultiplier = 2
  } = options

  let lastError: Error | null = null
  let retryCount = 0

  while (retryCount < maxRetries) {
    try {
      // 执行查询
      const result = await queryInternal(params)
      return result
    } catch (error) {
      lastError = error as Error

      // 分类错误
      const category = categorizeAPIError(lastError)

      if (category === null) {
        // 不可重试的错误直接抛出
        throw lastError
      }

      // 计算延迟
      let delay = baseDelay * Math.pow(backoffMultiplier, retryCount)
      delay = Math.min(delay, maxDelay)

      // 对于速率限制，使用更长的延迟
      if (category === 'rate_limit') {
        delay = Math.max(delay, 10000) // 至少 10 秒
      }

      console.log(
        `Query failed (${category}), retrying in ${delay}ms... (${retryCount + 1}/${maxRetries})`
      )

      // 等待后重试
      await sleep(delay)
      retryCount++
    }
  }

  throw lastError || new Error('Query failed after retries')
}
```

```typescript
export async function executeTool(
  toolUse: ToolUseBlock,
  context: ToolUseContext
): Promise<ToolResult> {
  const { name, id, input } = toolUse
  const startTime = Date.now()

  try {
    // 1. 查找工具
    const tool = findToolByName(name, context.allTools)
    if (!tool) {
      return {
        type: 'error',
        error: `Tool not found: ${name}`,
        toolId: id
      }
    }

    // 2. 权限检查
    const permission = await context.canUseTool(name)
    if (!permission.allowed) {
      return {
        type: 'error',
        error: `Tool denied: ${permission.reason}`,
        code: 'PERMISSION_DENIED',
        toolId: id
      }
    }

    // 3. 参数验证
    const validation = validateToolInput(input, tool.inputSchema)
    if (!validation.valid) {
      return {
        type: 'error',
        error: `Invalid input: ${validation.errors.join(', ')}`,
        code: 'INVALID_INPUT',
        toolId: id
      }
    }

    // 4. 执行工具（带超时）
    const timeout = getToolTimeout(name) // 工具特定的超时
    const result = await Promise.race([
      tool.invoke(input as Record<string, unknown>, context),
      sleep(timeout).then(() => {
        throw new Error(`Tool execution timeout after ${timeout}ms`)
      })
    ])

    // 5. 验证结果
    if (!isValidToolResult(result)) {
      return {
        type: 'error',
        error: 'Invalid tool result format',
        code: 'INVALID_RESULT',
        toolId: id
      }
    }

    // 6. 记录执行时间（用于分析）
    const duration = Date.now() - startTime
    logToolExecution({
      name,
      success: true,
      duration,
      inputSize: JSON.stringify(input).length,
      outputSize: JSON.stringify(result).length
    })

    return { ...result, toolId: id }

  } catch (error) {
    const duration = Date.now() - startTime

    // 分类错误
    const errorType = categorizeToolError(error)

    logToolExecution({
      name,
      success: false,
      duration,
      error: String(error),
      errorType
    })

    return {
      type: 'error',
      error: String(error),
      code: errorType,
      message: getErrorMessage(error),
      toolId: id
    }
  }
}

function categorizeToolError(error: unknown): string {
  if (error instanceof TimeoutError) return 'TIMEOUT'
  if (error instanceof PermissionError) return 'PERMISSION_DENIED'
  if (error instanceof ValidationError) return 'INVALID_INPUT'
  if (error instanceof NotFoundError) return 'NOT_FOUND'
  return 'EXECUTION_FAILED'
}
```

```typescript
  try {
    // 从持久化存储加载
    const stored = await getSessionStorage(sessionId)

    if (!stored) {
      return []
    }

    // 验证历史完整性
    const messages = stored.messages as Message[]

    // 检查是否有未完成的消息
    const lastMessage = messages[messages.length - 1]
    if (lastMessage && lastMessage.type === 'assistant') {
      // 检查是否被正确终止
      if (!hasValidStopReason(lastMessage)) {
        // 标记为被中断
        lastMessage.stopReason = 'stop_sequence'
      }
    }

    return messages
  } catch (error) {
    console.error('Failed to load conversation history:', error)
    return []
  }
}

function hasValidStopReason(msg: AssistantMessage): boolean {
  return ['end_turn', 'max_tokens', 'tool_use', 'stop_sequence'].includes(
    msg.stopReason || ''
  )
```

```typescript
export class TokenBudget {
  private inputTokens = 0
  private outputTokens = 0
  private maxTokens: number
  private warningThreshold: number
  private criticalThreshold: number

  constructor(maxTokens: number = 100_000) {
    this.maxTokens = maxTokens
    this.warningThreshold = maxTokens * 0.75
    this.criticalThreshold = maxTokens * 0.90
  }

  /**
   * 检查是否有足够的预算
   */
  canContinue(estimatedOutputTokens: number = 1000): boolean {
    const projected = this.inputTokens + this.outputTokens + estimatedOutputTokens
    return projected < this.maxTokens
  }

  /**
   * 获取剩余 token 数
   */
  getRemaining(): number {
    return Math.max(
      0,
      this.maxTokens - this.inputTokens - this.outputTokens
    )
  }

  /**
   * 报告使用量
   */
  reportUsage(
    inputTokens: number,
    outputTokens: number,
    cacheTokens: number = 0
  ): TokenBudgetStatus {
    this.inputTokens += inputTokens
    this.outputTokens += outputTokens

    const total = this.inputTokens + this.outputTokens
    const ratio = total / this.maxTokens

    return {
      total,
      remaining: this.getRemaining(),
      ratio,
      status: this.getStatus(ratio),
      shouldCompress: ratio > 0.7
    }
  }

  private getStatus(ratio: number): TokenBudgetStatus['status'] {
    if (ratio >= this.criticalThreshold) return 'critical'
    if (ratio >= this.warningThreshold) return 'warning'
    return 'normal'
  }
}
```
