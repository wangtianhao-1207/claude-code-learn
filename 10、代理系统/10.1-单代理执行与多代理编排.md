```typescript
export type AgentDefinition = {
  id: AgentId
  name: string
  description: string
  instructions: string              // 系统提示
  avatarEmoji?: string

  // 工具配置
  tools?: string[]                  // 允许的工具列表
  allowedTools?: string[]
  deniedTools?: string[]

  // 模型配置
  model?: string
  temperature?: number
  maxTokens?: number

  // 能力开关
  canUseInternet?: boolean
  canWriteFiles?: boolean
  canExecuteCode?: boolean

  // 元数据
  author?: string
  version?: string
  tags?: string[]
}

/**
 * 加载代理定义
 * 搜索路径优先级：
 * 1. 项目特定 (.claude/agents/{name}.json)
 * 2. 用户级别 (~/.claude-code/agents/{name}.json)
 * 3. 内置代理 (bundled/)
 */
export async function loadAgentDefinition(
  agentId: AgentId
): Promise<AgentDefinition | null> {
  const searchPaths = [
    // 项目级别
    join(getCwd(), '.claude', 'agents', `${agentId}.json`),
    // 用户级别
    join(getHomeDir(), '.claude-code', 'agents', `${agentId}.json`),
    // 内置
    join(__dirname, 'bundled', `${agentId}.json`)
  ]

  for (const path of searchPaths) {
    try {
      const content = await readFile(path, 'utf-8')
      return JSON.parse(content)
    } catch {
      // 继续下一个路径
    }
  }

  return null
}

/**
 * 列出所有可用代理
 */
export async function listAvailableAgents(): Promise<AgentDefinition[]> {
  const agents: AgentDefinition[] = []

  const searchDirs = [
    join(getCwd(), '.claude', 'agents'),
    join(getHomeDir(), '.claude-code', 'agents'),
    join(__dirname, 'bundled')
  ]

  for (const dir of searchDirs) {
    try {
      const files = await readdir(dir)
      for (const file of files) {
        if (file.endsWith('.json')) {
          const agentId = file.replace('.json', '')
          const agent = await loadAgentDefinition(agentId as AgentId)
          if (agent) {
            agents.push(agent)
          }
        }
      }
    } catch {
      // 目录不存在，继续
    }
  }

  return agents
}
```

## 2. AgentTool 实现

**文件**: `src/tools/AgentTool/AgentTool.ts:1-400`

```typescript
export const AgentTool: Tool = {
  name: 'agent',
  description: 'Spawn a sub-agent to handle a task',

  inputSchema: {
    type: 'object',
    properties: {
      agentId: {
        type: 'string',
        description: 'Agent to spawn'
      },
      task: {
        type: 'string',
        description: 'Task description'
      },
      maxIterations: {
        type: 'number',
        default: 10,
        description: 'Max iterations'
      },
      tools: {
        type: 'array',
        items: { type: 'string' },
        description: 'Override allowed tools'
      }
    },
    required: ['agentId', 'task']
  },

  async invoke(input, context) {
    const { agentId, task, maxIterations = 10, tools: overrideTools } = input

    try {
      // === 步骤 1: 加载代理定义 ===
      const agent = await loadAgentDefinition(String(agentId) as AgentId)

      if (!agent) {
        return {
          type: 'error',
          error: `Agent not found: ${agentId}`
        }
      }

      // === 步骤 2: 过滤工具 ===
      let allowedToolNames = overrideTools as string[] | undefined

      if (!allowedToolNames) {
        if (agent.allowedTools && agent.allowedTools.length > 0) {
          allowedToolNames = agent.allowedTools
        } else if (agent.deniedTools && agent.deniedTools.length > 0) {
          // 除了禁止的工具外，允许所有工具
          allowedToolNames = context.allTools
            .map(t => t.name)
            .filter(name => !agent.deniedTools!.includes(name))
        } else {
          // 允许所有工具
          allowedToolNames = context.allTools.map(t => t.name)
        }
      }

      const agentTools = context.allTools.filter(t =>
        allowedToolNames!.includes(t.name)
      )

      // === 步骤 3: 构建代理系统提示 ===
      const systemPrompt = `
You are ${agent.name}.

${agent.description}

Instructions:
${agent.instructions}

Task: ${task}

Available tools:
${agentTools.map(t => `- ${t.name}: ${t.description}`).join('\n')}

Complete this task using the available tools. Work efficiently and report your progress.
      `

      // === 步骤 4: 在子会话中执行 ===
      const result = await queryClaudeWithTools(
        task,
        systemPrompt,
        agentTools,
        {
          maxIterations,
          abortController: context.abortController,
          appState: context.appState,
          model: agent.model
        }
      )

      // === 步骤 5: 返回结果 ===
      return {
        type: 'text',
        text: `Agent ${agent.name} completed:\n\n${result.finalResponse}`
      }

    } catch (error) {
      return {
        type: 'error',
        error: String(error)
      }
    }
  }
}
```

## 3. 多代理协调

```typescript
/**
 * 协调器模式：中央协调器管理多个工作代理
 */

export async function runCoordinator(
  coordinatorDef: AgentDefinition,
  initialTask: string,
  context: ToolUseContext
): Promise<CoordinatorResult> {
  // === 步骤 1: 初始化协调器 ===
  const coordinatorPrompt = `
You are the coordinator for a team of specialized agents.

Your task: ${initialTask}

Available agents: [list of agent definitions]

Strategy:
1. Analyze the task
2. Break down into subtasks
3. Assign to appropriate agents
4. Collect results
5. Synthesize final answer

Be efficient and delegate work appropriately.
  `

  // === 步骤 2: 让协调器分析任务 ===
  const taskPlan = await queryClaudeWithTools(
    initialTask,
    coordinatorPrompt,
    context.allTools,
    { maxIterations: 5 }
  )

  // === 步骤 3: 解析子任务 ===
  const subtasks = parseSubtasks(taskPlan.finalResponse)

  // === 步骤 4: 并行执行子任务 ===
  const results = await Promise.all(
    subtasks.map(async (subtask) => {
      const agent = await loadAgentDefinition(subtask.agentId)

      if (!agent) {
        return { agentId: subtask.agentId, error: 'Agent not found' }
      }

      try {
        const result = await queryClaudeWithTools(
          subtask.description,
          `You are ${agent.name}. ${agent.instructions}`,
          context.allTools,
          { maxIterations: 10 }
        )

        return {
          agentId: subtask.agentId,
          result: result.finalResponse
        }
      } catch (error) {
        return {
          agentId: subtask.agentId,
          error: String(error)
        }
      }
    })
  )

  // === 步骤 5: 合成最终答案 ===
  const synthesisPrompt = `
Results from agents:
${results.map(r =>
  r.error
    ? `- ${r.agentId}: ERROR - ${r.error}`
    : `- ${r.agentId}: ${r.result}`
).join('\n')}

Synthesize these results into a cohesive answer to: ${initialTask}
  `

  const finalResult = await queryClaudeWithTools(
    synthesisPrompt,
    coordinatorPrompt,
    context.allTools,
    { maxIterations: 5 }
  )

  return {
    success: true,
    finalResponse: finalResult.finalResponse,
    subtasksCompleted: results.length,
    subtaskResults: results
  }
}

interface CoordinatorResult {
  success: boolean
  finalResponse: string
  subtasksCompleted: number
  subtaskResults: Array<{
    agentId: string
    result?: string
    error?: string
  }>
}
```

## 关键特性

✅ **灵活代理定义** - JSON 配置，轻松定制
✅ **工具过滤** - 每个代理可控制工具访问权限
✅ **递归代理** - 代理可生成子代理
✅ **并行执行** - 多个代理同时工作
✅ **中央协调** - 协调器管理复杂工作流
