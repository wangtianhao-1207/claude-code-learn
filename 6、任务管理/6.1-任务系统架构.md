# 6.1 任务系统架构与异步执行

## 任务系统概述

Claude Code 的任务系统支持后台异步执行，允许用户发起长时间运行的操作而不阻塞主对话流。

### 核心概念

```
Task Lifecycle:
pending → running → completed
       → running → failed
       → running → killed
```

## 2. 本地 Shell 任务实现

**文件**: `src/tasks/LocalShellTask/index.ts:1-500`

```typescript
export class LocalShellTask {
  private process: ChildProcess | null = null
  private outputStream: WriteStream | null = null

  /**
   * 启动 Shell 任务
   */
  async spawn(
    input: {
      command: string
      cwd?: string
      timeout?: number
      env?: Record<string, string>
    },
    context: TaskContext
  ): Promise<TaskHandle> {
    const taskId = generateTaskId('local_bash')
    const outputPath = getTaskOutputPath(taskId)

    try {
      // === 步骤 1: 创建输出流 ===
      await ensureDir(path.dirname(outputPath))
      this.outputStream = createWriteStream(outputPath, { flags: 'a' })

      // === 步骤 2: 启动子进程 ===
      this.process = spawn('bash', ['-c', input.command], {
        cwd: input.cwd || process.cwd(),
        env: { ...process.env, ...input.env },
        stdio: ['ignore', 'pipe', 'pipe'],
        detached: false
      })

      // === 步骤 3: 捕获输出 ===
      let totalOutputSize = 0
      const MAX_OUTPUT_SIZE = 10 * 1024 * 1024 // 10MB

      this.process.stdout?.on('data', (chunk: Buffer) => {
        totalOutputSize += chunk.length

        // 防止输出过大
        if (totalOutputSize > MAX_OUTPUT_SIZE) {
          this.process?.kill('SIGTERM')
        }

        this.outputStream?.write(chunk)

        // 报告进度
        context.onProgress?.({
          type: 'bash',
          status: 'running',
          outputSize: totalOutputSize
        })
      })

      this.process.stderr?.on('data', (chunk: Buffer) => {
        this.outputStream?.write(chunk)
      })

      // === 步骤 4: 处理完成 ===
      return new Promise((resolve, reject) => {
        const timeout = input.timeout
          ? setTimeout(() => {
              console.log(`Task timeout, killing process: ${taskId}`)
              this.process?.kill('SIGTERM')
            }, input.timeout)
          : null

        this.process!.on('exit', (code, signal) => {
          if (timeout) clearTimeout(timeout)

          this.outputStream?.end()

          resolve({
            taskId,
            exitCode: code ?? 1,
            outputPath,
            cleanup: () => {
              this.process?.kill('SIGKILL')
            }
          })
        })

        this.process!.on('error', (error) => {
          reject(error)
        })
      })
    } catch (error) {
      throw new Error(`Failed to spawn task: ${String(error)}`)
    }
  }

  /**
   * 中止任务
   */
  async kill(taskId: string): Promise<void> {
    if (this.process && !this.process.killed) {
      // 先发送 SIGTERM（温和）
      this.process.kill('SIGTERM')

      // 等待 5 秒
      await sleep(5000)

      // 若仍未结束，发送 SIGKILL（强制）
      if (!this.process.killed) {
        this.process.kill('SIGKILL')
      }
    }
  }
}

export type TaskHandle = {
  taskId: string
  exitCode: number
  outputPath: string
  cleanup: () => void
}
```

## 3. 任务管理和状态追踪

```typescript
// src/services/tasks/TaskManager.ts

export class TaskManager {
  private tasks = new Map<string, Task>()
  private taskHandles = new Map<string, TaskHandle>()

  /**
   * 创建新任务
   */
  async createTask(
    type: TaskType,
    description: string,
    context: TaskContext
  ): Promise<string> {
    const taskId = generateTaskId(type)

    const task: Task = {
      id: taskId,
      type,
      status: 'pending',
      description,
      createdAt: Date.now(),
      outputPath: getTaskOutputPath(taskId)
    }

    this.tasks.set(taskId, task)

    // 通知 UI
    context.setAppState((prev) => ({
      ...prev,
      tasks: new Map(prev.tasks).set(taskId, task)
    }))

    return taskId
  }

  /**
   * 启动任务
   */
  async startTask(
    taskId: string,
    context: TaskContext
  ): Promise<void> {
    const task = this.tasks.get(taskId)
    if (!task) {
      throw new Error(`Task not found: ${taskId}`)
    }

    // 更新状态为 running
    task.status = 'running'
    task.startedAt = Date.now()

    this.tasks.set(taskId, task)

    // 根据任务类型执行
    switch (task.type) {
      case 'local_bash':
        await this.runLocalBashTask(taskId, context)
        break

      case 'local_agent':
        await this.runLocalAgentTask(taskId, context)
        break

      // ... 其他任务类型
    }
  }

  /**
   * 获取任务输出
   */
  async getTaskOutput(taskId: string): Promise<string | null> {
    const task = this.tasks.get(taskId)
    if (!task) return null

    try {
      return await readFile(task.outputPath, 'utf-8')
    } catch {
      return null
    }
  }

  /**
   * 中止任务
   */
  async stopTask(taskId: string): Promise<void> {
    const task = this.tasks.get(taskId)
    if (!task) {
      throw new Error(`Task not found: ${taskId}`)
    }

    if (task.status !== 'running') {
      return
    }

    // 发送中止信号
    const handle = this.taskHandles.get(taskId)
    if (handle) {
      handle.cleanup?.()
    }

    // 更新状态
    task.status = 'killed'
    task.completedAt = Date.now()

    this.tasks.set(taskId, task)
  }
}
```

## 关键特性

✅ **后台执行** - 不阻塞主对话
✅ **进度报告** - 实时更新执行进度
✅ **输出持久化** - 保存到文件
✅ **错误恢复** - 支持重试和清理
✅ **资源管理** - 自动清理子进程
