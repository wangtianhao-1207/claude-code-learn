# 8.1 AppState 设计与状态管理

## 概述

AppState 是 Claude Code 的全局状态容器，实现单一真实源（SSOT）原则。所有 UI 状态都通过它驱动。

## 2. 状态管理实现

**文件**: `src/state/AppStateStore.ts:500-800`

```typescript
import create from 'zustand'
import { persist } from 'zustand/middleware'

/**
 * 创建 Zustand 状态存储
 */
export const createAppStateStore = (
  initialState: AppState,
  onStateChange?: (args: { newState: AppState; oldState: AppState }) => void
) => {
  return create<AppStateStore>()(
    persist(
      (set, get) => ({
        // 获取当前状态
        getState: () => get(),

        // 设置状态（支持函数和对象）
        setState: (updater) => {
          set((prev) => {
            const newState =
              typeof updater === 'function' ? updater(prev) : updater

            // 通知观察者
            if (onStateChange) {
              onStateChange({ newState, oldState: prev })
            }

            return newState
          })
        },

        // 选择器订阅（避免不必要重新渲染）
        subscribe: (selector, callback) => {
          let prevSelected = selector(get())

          return set((state) => {
            const newSelected = selector(state)

            // 只有选定值变化时才调用 callback
            if (!Object.is(prevSelected, newSelected)) {
              callback(newSelected)
            }

            prevSelected = newSelected
            return state
          })
        }
      }),
      {
        name: 'claude-code-state', // localStorage key
        version: 1,
        // 只持久化特定字段
        partialize: (state) => ({
          settings: state.settings,
          theme: state.theme,
          recentFiles: state.recentFiles
        })
      }
    )
  )
}

export type AppStateStore = {
  getState: () => AppState
  setState: (updater: ((prev: AppState) => AppState) | Partial<AppState>) => void
  subscribe: <T>(selector: (state: AppState) => T, callback: (value: T) => void) => Unsubscribe
}
```

```typescript
// ✅ 正确：使用纯函数更新（可追踪、可撤销）
setAppState((prev) => ({

  messages: s.messages,
  isWaiting: s.isWaitingForResponse

import { shallow } from 'zustand/react/shallow'

  s => ({ messages: s.messages, isWaiting: s.isWaitingForResponse }),
```

## 4. 状态持久化

```typescript
// src/utils/sessionStorage.ts

/**
 * 会话级状态存储
 */
export class SessionStorage {
  private sessionId: string

  constructor(sessionId: string) {
    this.sessionId = sessionId
  }

  /**
   * 保存状态快照
   */
  async saveSnapshot(appState: AppState): Promise<void> {
    const path = this.getSnapshotPath()

    const snapshot = {
      version: 1,
      timestamp: Date.now(),
      sessionId: this.sessionId,
      state: {
        messages: appState.messages,
        settings: appState.settings,
        tokenUsage: appState.tokenUsage,
        tasks: Array.from(appState.tasks.entries()),
        recentFiles: appState.recentFiles,
        cwd: appState.cwd
      }
    }

    await writeFile(path, JSON.stringify(snapshot, null, 2))
  }

  /**
   * 加载状态快照
   */
  async loadSnapshot(): Promise<Partial<AppState> | null> {
    try {
      const path = this.getSnapshotPath()
      const content = await readFile(path, 'utf-8')
      const snapshot = JSON.parse(content)

      // 验证版本
      if (snapshot.version !== 1) {
        console.warn('Snapshot version mismatch')
        return null
      }

      // 验证时间戳（超过 30 天则丢弃）
      const age = Date.now() - snapshot.timestamp
      if (age > 30 * 24 * 60 * 60 * 1000) {
        console.log('Snapshot too old, discarding')
        return null
      }

      return snapshot.state
    } catch (error) {
      console.error('Failed to load snapshot:', error)
      return null
    }
  }

  private getSnapshotPath(): string {
    return join(
      getHomeDir(),
      '.claude-code',
      'sessions',
      this.sessionId,
      'snapshot.json'
    )
  }
}
```

## 5. 状态监听和反应

```typescript
// src/state/onChangeAppState.ts

/**
 * 监听状态变化，执行副作用
 */
export function setupStateListeners(
  appState: AppState,
  store: AppStateStore
) {
  // === 监听消息变化 ===
  store.subscribe(
    (s) => s.messages.length,
    (messageCount) => {
      // 自动保存到会话存储
      saveSessionCheckpoint(appState.sessionId)

      // 更新消息索引
      updateMessageIndex(appState.messages)
    }
  )

  // === 监听 Token 使用 ===
  store.subscribe(
    (s) => s.tokenUsage.ratio,
    (ratio) => {
      if (ratio > 0.85) {
        // 显示警告
        console.warn('Token usage critical!')

        // 尝试自动压缩
        autoCompressMessages(appState)
      }
    }
  )

  // === 监听文件变化 ===
  store.subscribe(
    (s) => s.openFiles,
    (files) => {
      // 更新文件缓存统计
      updateFileCacheStats(files)
    }
  )

  // === 监听任务完成 ===
  store.subscribe(
    (s) => Array.from(s.tasks.values()).filter(t => t.status === 'completed').length,
    (completedCount) => {
      // 触发自动清理
      cleanupCompletedTasks()
    }
  )
}
```
