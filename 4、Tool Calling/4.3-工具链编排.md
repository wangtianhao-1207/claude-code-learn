# 4.3 工具链编排与复杂工作流

## 工具链概述

Claude Code 支持复杂的工具链编排，允许多个工具按照特定的依赖关系和逻辑顺序执行，形成完整的自动化工作流。

```typescript
/**
 * 工具链定义
 */
export interface ToolChainDefinition {
  // === 基本信息 ===
  id: string
  name: string
  description: string
  version: string

  // === 工具步骤 ===
  steps: ToolStep[]

  // === 全局配置 ===
  timeout?: number  // 整个链的超时时间
  retryPolicy?: RetryPolicy
  errorHandling?: ErrorHandlingStrategy
}

/**
 * 工具步骤定义
 */
export interface ToolStep {
  id: string
  toolName: string
  description: string

  // === 输入 ===
  input: Record<string, unknown> | InputTemplate

  // === 依赖关系 ===
  dependsOn?: string[]  // 依赖的步骤 ID
  condition?: string    // 条件表达式（如 "$.previous.success == true"）

  // === 错误处理 ===
  onError?: ErrorHandler
  retries?: number
  timeout?: number

  // === 输出 ===
  outputMapping?: Record<string, string>  // 提取结果的字段映射
}

/**
 * 输入模板（支持从前面步骤引用输出）
 *
 * 例如：
 * {
 *   "file_path": "${steps.read_file.result.path}",
 *   "pattern": "TODO"
 * }
 */
export type InputTemplate = Record<string, string | number | boolean>

/**
 * 重试策略
 */
export interface RetryPolicy {
  maxRetries: number
  backoffMs: number  // 指数退避基数
  retryableErrors?: string[]  // 可重试的错误类型
}

/**
 * 错误处理策略
 */
export type ErrorHandlingStrategy =
  | 'fail'       // 立即失败
  | 'continue'   // 继续执行（跳过此步骤）
  | 'fallback'   // 使用备用步骤

/**
 * 错误处理器
 */
export interface ErrorHandler {
  strategy: ErrorHandlingStrategy
  fallbackStepId?: string
  notifyUser?: boolean
  logError?: boolean
}
```

## 2. 工具链执行引擎

**文件**: `src/services/toolchain/ToolChainExecutor.ts:1-400`

```typescript
/**
 * 工具链执行器
 */
export class ToolChainExecutor {
  private definition: ToolChainDefinition
  private tools: Tool[]
  private context: ToolUseContext
  private stepResults = new Map<string, ToolResult>()
  private executionGraph: ExecutionGraph

  constructor(
    definition: ToolChainDefinition,
    tools: Tool[],
    context: ToolUseContext
  ) {
    this.definition = definition
    this.tools = tools
    this.context = context
    this.executionGraph = this.buildExecutionGraph()
  }

  /**
   * 构建执行图
   *
   * 将步骤依赖关系转换为有向无环图 (DAG)
   */
  private buildExecutionGraph(): ExecutionGraph {
    const graph: ExecutionGraph = {
      nodes: new Map(),
      adjacency: new Map()
    }

    // === 创建节点 ===
    for (const step of this.definition.steps) {
      graph.nodes.set(step.id, step)
      graph.adjacency.set(step.id, [])
    }

    // === 创建边（依赖关系） ===
    for (const step of this.definition.steps) {
      if (step.dependsOn) {
        for (const depId of step.dependsOn) {
          // depId → step.id （从前一步指向当前步）
          const deps = graph.adjacency.get(depId) || []
          deps.push(step.id)
          graph.adjacency.set(depId, deps)
        }
      }
    }

    return graph
  }

  /**
   * 执行工具链
   */
  async execute(): Promise<ChainExecutionResult> {
    const startTime = Date.now()

    try {
      // === 拓扑排序获取执行顺序 ===
      const executionOrder = this.topologicalSort()

      // === 执行步骤 ===
      const results: StepResult[] = []
      for (const stepId of executionOrder) {
        const step = this.definition.steps.find(s => s.id === stepId)!
        const result = await this.executeStep(step)
        results.push(result)

        // 检查是否需要中止
        if (result.status === 'failed' && step.onError?.strategy === 'fail') {
          return {
            success: false,
            steps: results,
            error: result.error,
            duration: Date.now() - startTime
          }
        }
      }

      return {
        success: true,
        steps: results,
        duration: Date.now() - startTime
      }

    } catch (error) {
      return {
        success: false,
        steps: Array.from(this.stepResults.entries()).map(([id, result]) => ({
          stepId: id,
          status: 'completed' as const,
          result
        })),
        error: String(error),
        duration: Date.now() - startTime
      }
    }
  }

  /**
   * 执行单个步骤
   */
  private async executeStep(step: ToolStep): Promise<StepResult> {
    const startTime = Date.now()

    try {
      // === 1. 检查条件 ===
      if (step.condition) {
        if (!this.evaluateCondition(step.condition)) {
          return {
            stepId: step.id,
            status: 'skipped' as const,
            duration: Date.now() - startTime
          }
        }
      }

      // === 2. 解析输入（处理模板变量） ===
      const input = this.resolveInput(step.input)

      // === 3. 查找工具 ===
      const tool = this.tools.find(t => t.name === step.toolName)
      if (!tool) {
        throw new Error(`Tool not found: ${step.toolName}`)
      }

      // === 4. 执行工具（带重试） ===
      let result: ToolResult | null = null
      let lastError: Error | null = null

      const maxAttempts = (step.retries ?? 0) + 1

      for (let attempt = 0; attempt < maxAttempts; attempt++) {
        try {
          result = await this.executeToolWithTimeout(
            tool,
            input,
            step.timeout ?? this.definition.timeout ?? 30000
          )

          if (result.type !== 'error') {
            break  // 成功，退出重试循环
          }

          lastError = new Error(result.error)

          // 指数退避重试
          if (attempt < maxAttempts - 1) {
            const backoff = (step.retries ?
              Math.pow(2, attempt) * 1000 :
              0)
            await sleep(backoff)
          }

        } catch (error) {
          lastError = error as Error
        }
      }

      if (!result) {
        throw lastError || new Error('Tool execution failed')
      }

      // === 5. 保存结果 ===
      this.stepResults.set(step.id, result)

      // === 6. 提取输出 ===
      const mappedOutput = this.mapOutput(result, step.outputMapping)

      return {
        stepId: step.id,
        status: 'completed' as const,
        result,
        output: mappedOutput,
        duration: Date.now() - startTime
      }

    } catch (error) {
      const stepError = error as Error

      // === 错误处理 ===
      if (step.onError?.strategy === 'fallback' && step.onError.fallbackStepId) {
        // 执行备用步骤
        const fallbackStep = this.definition.steps.find(
          s => s.id === step.onError?.fallbackStepId
        )
        if (fallbackStep) {
          return this.executeStep(fallbackStep)
        }
      }

      if (step.onError?.notifyUser) {
        console.error(`Step ${step.id} failed:`, stepError.message)
      }

      return {
        stepId: step.id,
        status: 'failed' as const,
        error: stepError.message,
        duration: Date.now() - startTime
      }
    }
  }

  /**
   * 执行工具（带超时保护）
   */
  private async executeToolWithTimeout(
    tool: Tool,
    input: Record<string, unknown>,
    timeoutMs: number
  ): Promise<ToolResult> {
    return Promise.race([
      tool.invoke(input, this.context),
      new Promise<ToolResult>((_, reject) =>
        setTimeout(() => reject(new Error('Tool timeout')), timeoutMs)
      )
    ])
  }

  /**
   * 拓扑排序
   *
   * 使用 Kahn 算法获取依赖的执行顺序
   */
  private topologicalSort(): string[] {
    const inDegree = new Map<string, number>()
    const queue: string[] = []
    const result: string[] = []

    // === 初始化入度 ===
    for (const [nodeId] of this.executionGraph.nodes) {
      inDegree.set(nodeId, 0)
    }

    for (const [fromId, toIds] of this.executionGraph.adjacency) {
      for (const toId of toIds) {
        inDegree.set(toId, (inDegree.get(toId) ?? 0) + 1)
      }
    }

    // === 找到所有入度为 0 的节点 ===
    for (const [nodeId, degree] of inDegree) {
      if (degree === 0) {
        queue.push(nodeId)
      }
    }

    // === Kahn 算法 ===
    while (queue.length > 0) {
      const nodeId = queue.shift()!
      result.push(nodeId)

      // 处理所有依赖这个节点的节点
      const dependents = this.executionGraph.adjacency.get(nodeId) || []
      for (const dependentId of dependents) {
        inDegree.set(dependentId, inDegree.get(dependentId)! - 1)
        if (inDegree.get(dependentId) === 0) {
          queue.push(dependentId)
        }
      }
    }

    // === 检查循环 ===
    if (result.length !== this.executionGraph.nodes.size) {
      throw new Error('Circular dependency detected in tool chain')
    }

    return result
  }

  /**
   * 条件求值
   *
   * 支持简单的 JSONPath 表达式
   */
  private evaluateCondition(condition: string): boolean {
    try {
      // 简化实现：支持 $.stepId.field == value
      const match = condition.match(/\$\.(\w+)\.(\w+)\s*==\s*(.+)/)
      if (!match) return true

      const [, stepId, field, expected] = match
      const result = this.stepResults.get(stepId)

      if (!result || result.type !== 'text') return false

      try {
        const data = JSON.parse(result.text)
        const value = data[field]
        return String(value) === expected.trim().replace(/['"]/g, '')
      } catch {
        return false
      }

    } catch (error) {
      console.error('Error evaluating condition:', error)
      return false
    }
  }

  /**
   * 解析输入（处理模板变量）
   *
   * 例如：${steps.step1.result.path} 会被替换为 step1 的输出中的 path 字段
   */
  private resolveInput(
    input: Record<string, unknown> | InputTemplate
  ): Record<string, unknown> {
    const resolved: Record<string, unknown> = {}

    for (const [key, value] of Object.entries(input)) {
      if (typeof value === 'string' && value.includes('${steps.')) {
        // 解析模板变量
        resolved[key] = this.resolveTemplateVariable(value)
      } else {
        resolved[key] = value
      }
    }

    return resolved
  }

  /**
   * 解析模板变量
   *
   * 格式：${steps.stepId.result.path}
   */
  private resolveTemplateVariable(template: string): unknown {
    const match = template.match(/\$\{steps\.(\w+)\.(\w+)\.(.+?)\}/)
    if (!match) return template

    const [, stepId, prop, path] = match
    const stepResult = this.stepResults.get(stepId)

    if (!stepResult || stepResult.type !== 'text') {
      return null
    }

    try {
      const data = JSON.parse(stepResult.text)
      return getNestedValue(data, path.split('.'))
    } catch {
      return null
    }
  }

  /**
   * 映射输出
   */
  private mapOutput(
    result: ToolResult,
    mapping?: Record<string, string>
  ): Record<string, unknown> {
    if (!mapping || result.type !== 'text') {
      return {}
    }

    try {
      const data = JSON.parse(result.text)
      const mapped: Record<string, unknown> = {}

      for (const [key, path] of Object.entries(mapping)) {
        mapped[key] = getNestedValue(data, path.split('.'))
      }

      return mapped
    } catch {
      return {}
    }
  }
}

/**
 * 执行图
 */
interface ExecutionGraph {
  nodes: Map<string, ToolStep>
  adjacency: Map<string, string[]>
}

/**
 * 步骤执行结果
 */
interface StepResult {
  stepId: string
  status: 'completed' | 'failed' | 'skipped'
  result?: ToolResult
  output?: Record<string, unknown>
  error?: string
  duration: number
}

/**
 * 链执行结果
 */
export interface ChainExecutionResult {
  success: boolean
  steps: StepResult[]
  error?: string
  duration: number
}

/**
 * 辅助函数：获取嵌套值
 */
function getNestedValue(
  obj: Record<string, unknown>,
  path: string[]
): unknown {
  let current: unknown = obj
  for (const key of path) {
    if (typeof current === 'object' && current !== null) {
      current = (current as Record<string, unknown>)[key]
    } else {
      return undefined
    }
  }
  return current
}
```

```yaml

id: code_review_workflow
name: Code Review Workflow
description: Review recent code changes and provide feedback

steps:
  # 步骤 1: 获取 Git diff
  - id: get_diff
    toolName: bash
    description: Get recent code changes
    input:
      command: git diff HEAD~1
    timeout: 10000

  # 步骤 2: 读取关键文件
  - id: read_package_json
    toolName: read
    description: Read package.json to understand dependencies
    input:
      file_path: package.json
    dependsOn:
      - get_diff

  # 步骤 3: 分析代码质量（仅在有改动时）
  - id: analyze_quality
    toolName: bash
    description: Run ESLint
    input:
      command: npx eslint . --format json
    dependsOn:
      - get_diff
    condition: "$.get_diff.result.lines_changed > 0"
    retries: 2
    timeout: 30000

  # 步骤 4: 使用 Claude 生成审查意见
  - id: generate_review
    toolName: agent
    description: Generate code review using Claude
    input:
      agentId: code_reviewer
      task: "Review the following code changes:\n\nDiff:\n${steps.get_diff.result.content}\n\nLint results:\n${steps.analyze_quality.result.issues}"
    dependsOn:
      - analyze_quality
    onError:
      strategy: continue
      notifyUser: true

  # 步骤 5: 保存审查结果
  - id: save_review
    toolName: write
    description: Save review to file
    input:
      file_path: review_${timestamp}.md
      content: |
        # Code Review

        ## Changes
        ${steps.get_diff.result.summary}

        ## Analysis
        ${steps.analyze_quality.result.summary}

        ## Review
        ${steps.generate_review.result.feedback}
    dependsOn:
      - generate_review
    onError:
      strategy: fail

timeout: 120000
retryPolicy:
  maxRetries: 3
  backoffMs: 1000
  retryableErrors:
    - timeout
    - network_error

errorHandling: fail
```

```typescript
/**
 * 代码生成工作流示例

async function runCodeGenerationWorkflow(feature: string) {
  const chainDef: ToolChainDefinition = {
    id: 'code_generation',
    name: 'Feature Code Generation',
    description: 'Generate code for a new feature',
    version: '1.0.0',
    timeout: 300000,  // 5 分钟

    steps: [
      // 1. 分析需求
      {
        id: 'analyze_requirement',
        toolName: 'agent',
        description: 'Analyze feature requirement',
        input: {
          agentId: 'requirements_analyzer',
          task: feature
        }
      },

      // 2. 查找相关代码示例
      {
        id: 'find_examples',
        toolName: 'glob',
        description: 'Find similar code patterns',
        input: {
          pattern: 'src/**/*.ts',
          path: '.'
        },
        dependsOn: ['analyze_requirement']
      },

      // 3. 读取关键文件
      {
        id: 'read_architecture',
        toolName: 'read',
        description: 'Read architecture documentation',
        input: {
          file_path: 'docs/architecture.md'
        }
      },

      // 4. 生成代码
      {
        id: 'generate_code',
        toolName: 'agent',
        description: 'Generate implementation code',
        input: {
          agentId: 'code_generator',
          task: `Generate code based on:
Requirements: ${steps.analyze_requirement.result}
Examples: ${steps.find_examples.result}
Architecture: ${steps.read_architecture.result}`
        },
        dependsOn: ['analyze_requirement', 'read_architecture'],
        timeout: 120000
      },

      // 5. 添加测试
      {
        id: 'generate_tests',
        toolName: 'agent',
        description: 'Generate unit tests',
        input: {
          agentId: 'test_generator',
          task: `Generate tests for:
${steps.generate_code.result}`
        },
        dependsOn: ['generate_code']
      },

      // 6. 保存文件
      {
        id: 'save_implementation',
        toolName: 'write',
        description: 'Save generated code',
        input: {
          file_path: 'src/generated/${feature}.ts',
          content: '${steps.generate_code.result.code}'
        },
        dependsOn: ['generate_code']
      },

      {
        id: 'save_tests',
        toolName: 'write',
        description: 'Save generated tests',
        input: {
          file_path: 'tests/${feature}.test.ts',
          content: '${steps.generate_tests.result.tests}'
        },
        dependsOn: ['generate_tests']
      },

      // 7. 运行测试
      {
        id: 'run_tests',
        toolName: 'bash',
        description: 'Run tests to verify',
        input: {
          command: 'npm test -- ${feature}.test.ts'
        },
        dependsOn: ['save_tests'],
        retries: 2,
        onError: {
          strategy: 'fail',
          notifyUser: true
        }
      }
    ]
  }

  // 执行工具链
  const executor = new ToolChainExecutor(chainDef, allTools, context)
  const result = await executor.execute()

  return result
}
```

## 关键特性

✅ **DAG 执行** - 支持复杂的依赖关系，自动并行执行无依赖步骤
✅ **条件执行** - 支持条件判断，动态选择执行路径
✅ **模板变量** - 支持引用前面步骤的输出作为输入
✅ **重试机制** - 自动重试失败的步骤（指数退避）
✅ **错误处理** - fail/continue/fallback 多种错误处理策略
✅ **超时保护** - 支持全局和步骤级的超时控制
✅ **输出映射** - 灵活的输出提取和转换
✅ **执行监控** - 详细的执行结果和性能指标
