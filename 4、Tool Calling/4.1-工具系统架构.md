# 4.1 工具系统架构与注册机制

## 工具系统概述

Claude Code 的工具系统采用插件架构，支持内置工具、MCP 工具、自定义工具的统一调用。每个工具通过标准接口定义，经过权限检查后执行。

```typescript
/**
 * 工具接口（核心定义）
 */
export interface Tool {
  // === 基本信息 ===
  name: string                          // 工具唯一标识（如 'bash', 'read'）
  description: string                   // 工具描述（显示给 Claude）
  category?: 'file' | 'bash' | 'web' | 'network' | 'custom'

  // === 输入定义（JSON Schema） ===
  inputSchema: {
    type: 'object'
    properties: Record<string, unknown>  // 参数定义
    required?: string[]                  // 必需参数
    additionalProperties?: boolean
  }

  // === 执行函数 ===
  invoke(
    input: Record<string, unknown>,      // 参数对象
    context: ToolUseContext              // 执行上下文
  ): Promise<ToolResult>

  // === 可选：权限控制 ===
  requiredPermissions?: Permission[]     // 必需的权限
  minTokenBudget?: number                // 最小 Token 预算
  timeout?: number                       // 超时时间（ms）
  maxConcurrency?: number                // 最大并发数

  // === 可选：元数据 ===
  tags?: string[]                        // 标签（用于分类和搜索）
  deprecated?: boolean                   // 是否已弃用
  version?: string                       // 版本号
}

/**
 * 工具执行上下文
 */
export interface ToolUseContext {
  // === 应用状态 ===
  appState: AppState                     // 全局应用状态
  setAppState: (update: AppState) => void

  // === 权限检查 ===
  canUseTool: (name: string) => boolean
  checkPermission: (action: string) => Promise<boolean>

  // === 缓存和资源 ===
  readFileCache?: Map<string, string>    // 文件读取缓存
  reportMetrics?: (metrics: ToolMetrics) => void

  // === 执行控制 ===
  abortController?: AbortController      // 中止信号
  timeoutMs?: number                     // 执行超时

  // === 其他 ===
  allTools: Tool[]                       // 所有可用工具列表
}

/**
 * 工具执行结果
 */
export type ToolResult =
  | TextResult                           // 文本结果
  | ToolError                            // 错误结果
  | BinaryResult                         // 二进制结果

export interface TextResult {
  type: 'text'
  text: string
  metadata?: {
    lineCount?: number
    isPartial?: boolean                  // 是否被截断
    truncatedFrom?: number               // 原始大小
  }
}

export interface ToolError {
  type: 'error'
  error: string
  errorCode?: string                     // 错误代码
  details?: Record<string, unknown>
}

export interface BinaryResult {
  type: 'binary'
  mimeType: string
  data: Buffer
  size: number
}

/**
 * 工具指标
 */
export interface ToolMetrics {
  toolName: string
  executionTime: number                  // 执行时间（ms）
  inputTokens?: number                   // 输入 Token 数
  outputTokens?: number                  // 输出 Token 数
  success: boolean
  errorCode?: string
}
```

## 3. 工具注册表

**文件**: `src/tools.ts:100-200`

```typescript
/**
 * 工具注册表（工厂模式）
 */
export class ToolRegistry {
  private tools = new Map<string, Tool>()

  /**
   * 构造函数：注册所有内置工具
   */
  constructor() {
    this.registerBuiltinTools()
  }

  /**
   * 注册内置工具
   */
  private registerBuiltinTools(): void {
    // 文件操作
    this.register(ReadTool)
    this.register(WriteTool)
    this.register(EditTool)
    this.register(GlobTool)
    this.register(GrepTool)

    // Shell 执行
    this.register(BashTool)
    this.register(GitTool)

    // 网络操作
    this.register(WebSearchTool)
    this.register(WebFetchTool)

    // 自定义工具
    this.register(MemoryTool)
    this.register(TaskTool)
    this.register(AgentTool)
  }

  /**
   * 注册工具
   */
  register(tool: Tool): void {
    if (this.tools.has(tool.name)) {
      console.warn(`Tool already registered: ${tool.name}`)
      return
    }

    // === 验证工具定义 ===
    validateToolDefinition(tool)

    this.tools.set(tool.name, tool)
    console.log(`✓ Registered tool: ${tool.name}`)
  }

  /**
   * 获取工具
   */
  getTool(name: string): Tool | undefined {
    return this.tools.get(name)
  }

  /**
   * 列出所有工具
   */
  listTools(): Tool[] {
    return Array.from(this.tools.values())
  }

  /**
   * 按分类筛选
   */
  getToolsByCategory(category: string): Tool[] {
    return Array.from(this.tools.values()).filter(
      t => t.category === category
    )
  }

  /**
   * 动态添加工具（用于插件）
   */
  addPluginTools(tools: Tool[]): void {
    for (const tool of tools) {
      this.register(tool)
    }
  }
}

/**
 * 验证工具定义
 */
function validateToolDefinition(tool: Tool): void {
  // === 必需字段 ===
  if (!tool.name || !/^[a-z0-9_-]+$/.test(tool.name)) {
    throw new Error(`Invalid tool name: ${tool.name}`)
  }

  if (!tool.description || tool.description.length < 10) {
    throw new Error(`Tool description too short: ${tool.name}`)
  }

  if (!tool.inputSchema) {
    throw new Error(`Missing inputSchema: ${tool.name}`)
  }

  // === 验证 JSON Schema ===
  if (tool.inputSchema.type !== 'object') {
    throw new Error(`inputSchema must be object type: ${tool.name}`)
  }

  if (!tool.inputSchema.properties) {
    throw new Error(`Missing schema properties: ${tool.name}`)
  }

  // === 验证执行函数 ===
  if (typeof tool.invoke !== 'function') {
    throw new Error(`Missing invoke function: ${tool.name}`)
  }

  // === 验证权限 ===
  if (tool.requiredPermissions) {
    for (const perm of tool.requiredPermissions) {
      if (!isValidPermission(perm)) {
        throw new Error(`Invalid permission: ${perm}`)
      }
    }
  }
}
```

## 4. 工具加载和初始化

**文件**: `src/services/tools/toolLoader.ts:1-150`

```typescript
/**
 * 工具加载器
 */
export class ToolLoader {
  private registry: ToolRegistry

  constructor() {
    this.registry = new ToolRegistry()
  }

  /**
   * 加载所有可用工具
   */
  async loadAllTools(): Promise<Tool[]> {
    const tools: Tool[] = []

    // === 1. 加载内置工具 ===
    tools.push(...this.registry.listTools())

    // === 2. 加载 MCP 工具 ===
    try {
      const mcpTools = await this.loadMCPTools()
      tools.push(...mcpTools)
    } catch (error) {
      console.warn('Failed to load MCP tools:', error)
    }

    // === 3. 加载自定义工具（从插件） ===
    try {
      const pluginTools = await this.loadPluginTools()
      tools.push(...pluginTools)
    } catch (error) {
      console.warn('Failed to load plugin tools:', error)
    }

    return tools
  }

  /**
   * 从 MCP 服务器加载工具
   */
  private async loadMCPTools(): Promise<Tool[]> {
    const tools: Tool[] = []
    const mcpClient = new MCPClient()

    // 获取已配置的 MCP 服务器
    const servers = await mcpClient.listServers()

    for (const server of servers) {
      try {
        const connection = await mcpClient.ensureConnected(server.id)

        // 获取服务器提供的工具
        for (const toolDef of connection.tools || []) {
          const tool: Tool = {
            name: `mcp:${server.id}:${toolDef.name}`,
            description: toolDef.description,
            category: 'custom',
            inputSchema: toolDef.inputSchema,

            async invoke(input, context) {
              return mcpClient.callTool(
                server.id,
                toolDef.name,
                input,
                context
              )
            }
          }

          tools.push(tool)
        }
      } catch (error) {
        console.warn(`Failed to load tools from MCP: ${server.id}`, error)
      }
    }

    return tools
  }

  /**
   * 从插件加载工具
   */
  private async loadPluginTools(): Promise<Tool[]> {
    const pluginDir = join(getHomeDir(), '.claude-code', 'plugins')
    const tools: Tool[] = []

    try {
      const files = await readdir(pluginDir)

      for (const file of files) {
        if (file.endsWith('.js')) {
          const pluginPath = join(pluginDir, file)
          const plugin = await import(pluginPath)

          if (plugin.tools && Array.isArray(plugin.tools)) {
            tools.push(...plugin.tools)
          }
        }
      }
    } catch (error) {
      // 插件目录可能不存在
    }

    return tools
  }
}

/**
 * 全局工具加载
 */
export async function loadTools(): Promise<Tool[]> {
  const loader = new ToolLoader()
  return loader.loadAllTools()
}
```

## 5. 工具动态查找

```typescript
/**
 * 根据名称查找工具
 */
export function findToolByName(
  name: string,
  tools: Tool[]
): Tool | undefined {
  return tools.find(t => t.name === name)
}

/**
 * 模糊搜索工具
 */
export function searchTools(
  query: string,
  tools: Tool[]
): Tool[] {
  const lowerQuery = query.toLowerCase()

  return tools.filter(t =>
    t.name.includes(lowerQuery) ||
    t.description.toLowerCase().includes(lowerQuery) ||
    (t.tags?.some(tag => tag.includes(lowerQuery)) ?? false)
  )
}

/**
 * 获取工具帮助信息
 */
export function getToolHelp(tool: Tool): string {
  const schema = tool.inputSchema as any

  let help = `# ${tool.name}\n\n`
  help += `${tool.description}\n\n`
  help += `## 参数\n\n`

  for (const [name, prop] of Object.entries(schema.properties || {})) {
    const required = schema.required?.includes(name) ?? false
    const requiredStr = required ? '(必需)' : '(可选)'
    help += `- **${name}** ${requiredStr}\n`
    help += `  - 类型: ${(prop as any).type}\n`
    help += `  - 描述: ${(prop as any).description}\n\n`
  }

  if (tool.timeout) {
    help += `## 超时\n${tool.timeout}ms\n\n`
  }

  if (tool.requiredPermissions) {
    help += `## 权限\n${tool.requiredPermissions.join(', ')}\n`
  }

  return help
}
```
