```typescript
// 工具执行完整流程图

/*
Tool Use Detected (API 响应)
         ↓
[Extract Tool Info]
  - name, id, input
         ↓
[Find Tool in Registry]
  ├─ 成功 → [Validate Input]
  └─ 失败 → ERROR: Tool not found
         ↓
[Validate Input]
  - 检查 JSON Schema
  ├─ 有效 → [Check Permissions]
  └─ 无效 → ERROR: Invalid input
         ↓
[Check Permissions]
  - 5 层权限检查
  ├─ 允许 → [Execute Tool]
  ├─ 拒绝 → ERROR: Permission denied
  └─ 需审批 → Show UI Prompt
         ↓
[Execute Tool]
  - 超时: 30 秒（可配置）
  - 并行度: 最多 20 个
  ├─ 成功 → [Format Result]
  ├─ 超时 → ERROR: Timeout
  └─ 异常 → ERROR: Execution failed
         ↓
[Format Result]
  - 截断大输出 (10KB)
  - 转换为 API 格式
         ↓
[Add to Message History]
  - type: tool_result
  - tool_use_id: 关联
  - is_error: 标记
         ↓
Continue Query Loop

/**
 * 并行执行多个工具
 */
export async function runToolsParallel(
  toolUses: ToolUseBlock[],
  context: ToolUseContext,
  iteration: number
): Promise<ToolResultBlockParam[]> {
  // 限制并发度
  const limiter = pLimit(20)

  // 创建执行任务
  const tasks = toolUses.map(toolUse =>
    limiter(() =>
      executeTool(toolUse, {
        ...context,
        iteration
      })
    )
  )

  // 并行等待
  return await Promise.all(tasks)
}

/**
 * 执行单个工具
 */
export async function executeTool(
  toolUse: ToolUseBlock,
  context: ToolUseContext
): Promise<ToolResultBlockParam> {
  const { name, id, input } = toolUse
  const startTime = performance.now()

  try {
    // === 1. 查找工具 ===
    const tool = findToolByName(name, context.allTools)
    if (!tool) {
      return formatToolError(
        id,
        `Tool not found: ${name}`,
        'NOT_FOUND'
      )
    }

    // === 2. 权限检查 ===
    const permission = await context.canUseTool(name, input)
    if (!permission.allowed) {
      return formatToolError(
        id,
        `Tool denied: ${permission.reason}`,
        'PERMISSION_DENIED'
      )
    }

    // === 3. 参数验证 ===
    const validation = validateToolInput(input, tool.inputSchema)
    if (!validation.valid) {
      const errors = validation.errors
        .map(e => `${e.path || 'root'}: ${e.message}`)
        .join('\n')
      return formatToolError(
        id,
        `Invalid input:\n${errors}`,
        'INVALID_INPUT'
      )
    }

    // === 4. 执行工具（带超时） ===
    const timeout = tool.estimateExecutionTime?.(input) || 30_000

    let result: ToolResult
    try {
      result = await Promise.race([
        tool.invoke(input as Record<string, unknown>, context),
        sleep(timeout).then(() => {
          throw new ToolTimeoutError(name, timeout)
        })
      ])
    } catch (error) {
      if (error instanceof ToolTimeoutError) {
        return formatToolError(
          id,
          `Tool execution timeout after ${timeout}ms`,
          'TIMEOUT'
        )
      }
      throw error
    }

    // === 5. 验证结果 ===
    if (!isValidToolResult(result)) {
      return formatToolError(
        id,
        'Tool returned invalid result format',
        'INVALID_RESULT'
      )
    }

    // === 6. 处理结果 ===
    const duration = performance.now() - startTime

    // 记录指标
    context.reportMetrics?.({
      toolName: name,
      success: true,
      duration,
      inputSize: JSON.stringify(input).length,
      outputSize: JSON.stringify(result).length,
      iteration: context.iteration
    })

    // 格式化并返回
    return {
      type: 'tool_result',
      tool_use_id: id,
      content: formatToolResultForAPI(result),
      is_error: false
    }

  } catch (error) {
    const duration = performance.now() - startTime

    // 分类错误
    const errorCode = categorizeToolError(error)

    context.reportMetrics?.({
      toolName: name,
      success: false,
      duration,
      error: String(error),
      errorCode
    })

    return formatToolError(
      id,
      String(error),
      errorCode
    )
  }
}

function formatToolError(
  toolUseId: string,
  message: string,
  code: string
): ToolResultBlockParam {
  return {
    type: 'tool_result',
    tool_use_id: toolUseId,
    content: [{
      type: 'text',
      text: `Error (${code}): ${message}`
    }],
    is_error: true
  }
}
```

```typescript
// src/tools/FileReadTool/index.ts:1-200
export const FileReadTool: Tool = {
  name: 'read',
  description: 'Read file contents',
  inputSchema: {
    type: 'object',
    properties: {
      file_path: { type: 'string', description: 'File path' },
      offset: { type: 'number', description: 'Byte offset' },
      limit: { type: 'number', description: 'Max bytes' }
    },
    required: ['file_path']
  },

  async invoke(input, context) {
    const filePath = String(input.file_path)
    const offset = Number(input.offset || 0)
    const limit = input.limit ? Number(input.limit) : undefined

    // 权限检查
    if (!isPathAllowed(filePath, context.appState.cwd)) {
      return {
        type: 'error',
        error: `Access denied: ${filePath}`
      }
    }

    try {
      // 使用缓存避免重复读取
      const cached = await context.readFileCache.getOrLoad(
        filePath,
        offset,
        limit
      )

      if (!cached) {
        return {
          type: 'error',
          error: `File not found: ${filePath}`
        }
      }

      return {
        type: 'text',
        text: cached.content
      }
    } catch (error) {
      return {
        type: 'error',
        error: String(error)
      }
    }
  }
}
```

### 2.2 BashTool

```typescript
// src/tools/BashTool/index.ts:1-250
export const BashTool: Tool = {
  name: 'bash',
  description: 'Execute shell commands',
  inputSchema: {
    type: 'object',
    properties: {
      command: { type: 'string' },
      cwd: { type: 'string' },
      timeout: { type: 'number', default: 30_000 }
    },
    required: ['command']
  },

  estimateExecutionTime(input) {
    const cmd = String(input.command || '')
    // npm 安装类命令需要更长时间
    if (cmd.match(/npm|yarn|pip.*install/)) {
      return 120_000 // 2 分钟
    }
    return 30_000 // 默认 30 秒
  },

  async invoke(input, context) {
    const command = String(input.command)
    const cwd = String(input.cwd || context.appState.cwd)
    const timeout = Number(input.timeout || 30_000)

    // 权限检查：禁止危险命令
    if (isDangerousCommand(command)) {
      return {
        type: 'error',
        error: `Dangerous command blocked: ${command.substring(0, 50)}`
      }
    }

    try {
      // 创建本地 Shell 任务
      const shellTask = new LocalShellTask()
      const taskHandle = await shellTask.spawn(
        { command, cwd, timeout },
        context as any
      )

      // 读取输出
      const output = await readFile(taskHandle.outputPath, 'utf-8')

      // 截断大输出
      const trimmed = trimToolOutput(output)

      return {
        type: 'text',
        text: trimmed
      }
    } catch (error) {
      return {
        type: 'error',
        error: String(error)
      }
    }
  }
}
```

### 2.3 FileEditTool

```typescript
// src/tools/FileEditTool/index.ts:1-300
export const FileEditTool: Tool = {
  name: 'edit',
  description: 'Edit file contents with precise replacements',
  inputSchema: {
    type: 'object',
    properties: {
      file_path: { type: 'string' },
      edits: {
        type: 'array',
        items: {
          type: 'object',
          properties: {
            old_string: { type: 'string' },
            new_string: { type: 'string' }
          },
          required: ['old_string', 'new_string']
        }
      }
    },
    required: ['file_path', 'edits']
  },

  async invoke(input, context) {
    const filePath = String(input.file_path)
    const edits = input.edits as Array<{ old_string: string; new_string: string }>

    try {
      // === 检查 1: 权限检查 ===
      if (!isPathAllowed(filePath, context.appState.cwd)) {
        return { type: 'error', error: `Access denied` }
      }

      // === 检查 2: 文件必须先被读取（缓存中存在） ===
      const cached = context.readFileCache.getOrLoad(filePath)
      if (!cached) {
        return {
          type: 'error',
          error: `File must be read before editing`
        }
      }

      // === 检查 3: 验证编辑可应用 ===
      let content = cached.content
      for (const edit of edits) {
        if (!content.includes(edit.old_string)) {
          return {
            type: 'error',
            error: `Pattern not found: ${edit.old_string.substring(0, 50)}...`
          }
        }
      }

      // === 步骤 4: 应用编辑 ===
      for (const edit of edits) {
        content = content.replace(edit.old_string, edit.new_string)
      }

      // === 步骤 5: 原子保存（带备份） ===
      const backupPath = `${filePath}.bak`
      await writeFile(backupPath, cached.content)
      await writeFile(filePath, content)

      // === 步骤 6: 清除缓存 ===
      context.readFileCache.invalidate(filePath)

      return {
        type: 'text',
        text: `Edited ${filePath} (${edits.length} changes)`
      }
    } catch (error) {
      return {
        type: 'error',
        error: String(error)
      }
    }
  }
}
```

```typescript
export const GlobTool: Tool = {
  name: 'glob',
  description: 'Find files matching glob pattern',
  inputSchema: {
    type: 'object',
    properties: {
      pattern: { type: 'string', description: 'Glob pattern' },
      ignore: { type: 'array', items: { type: 'string' } },
      limit: { type: 'number', default: 1000 }
    },
    required: ['pattern']
  },

  async invoke(input, context) {
    const pattern = String(input.pattern)
    const ignore = (input.ignore as string[]) || []
    const limit = Number(input.limit || 1000)

    try {
      // 使用 fast-glob 高效搜索
      const files = await glob(pattern, {
        cwd: context.appState.cwd,
        ignore,
        absolute: false,
        suppressErrors: true
      })

      // 限制结果数量
      const limited = files.slice(0, limit)

      return {
        type: 'text',
        text: limited.join('\n')
      }
    } catch (error) {
      return {
        type: 'error',
        error: String(error)
      }
    }
  }
}
```

```typescript
export const WebSearchTool: Tool = {

      return {
        type: 'error',
        error: String(error)
      }

  }
```

## 3. 工具执行的关键优化

### 3.1 缓存机制

```typescript
// 文件读取缓存（避免重复读取）
class FileStateCache {
  private cache = new LRUCache<string, FileState>({
    max: 100,              // 最多 100 个文件
    maxSize: 25 * 1024 * 1024  // 25MB
  })

  async getOrLoad(filePath: string): Promise<FileState | null> {
    const normalized = normalize(filePath)

    // 检查缓存
    if (this.cache.has(normalized)) {
      return this.cache.get(normalized)!
    }

    // 从磁盘读取
    try {
      const content = await readFile(filePath, 'utf-8')
      const state = {
        content,
        timestamp: Date.now(),
        hash: computeHash(content)
      }
      this.cache.set(normalized, state)
      return state
    } catch {
      return null
    }
  }

  hasChanged(filePath: string, newContent: string): boolean {
    const cached = this.cache.get(normalize(filePath))
    if (!cached) return true
    return computeHash(newContent) !== cached.hash
  }
}
```

### 3.2 工具输出处理

```typescript
// 截断、格式化、缓存工具结果
function formatToolResultForAPI(result: ToolResult): ContentBlockParam {
  switch (result.type) {
    case 'text': {
      // 截断过大输出
      let text = result.text
      if (text.length > 10_000) {
        const half = 5_000
        text = text.substring(0, half) +
          '\n... [output truncated] ...\n' +
          text.substring(text.length - half)
      }
      return { type: 'text', text }
    }

    case 'error':
      return {
        type: 'text',
        text: `Error: ${result.error}`
      }

    case 'streaming':
      // 流式结果需要实时处理
      return {
        type: 'text',
        text: '[Tool streamed output...]'
      }

    default:
      return {
        type: 'text',
        text: 'Unknown result type'
      }
  }
}
```

## 关键指标

- **工具查找**: < 10ms
- **权限检查**: < 50ms
- **输入验证**: < 20ms
- **执行平均**: 500-5000ms（取决于工具）
- **并发度**: 20 个工具同时执行
- **输出截断**: 10KB
