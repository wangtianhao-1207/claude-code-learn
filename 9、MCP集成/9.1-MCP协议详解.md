# 9.1 MCP 协议与集成详解

## MCP 的核心价值

Model Context Protocol (MCP) 是 Claude Code 连接外部工具和服务的标准化协议。

**关键优势**：
- **标准化**：任何工具都可以实现 MCP
- **互操作**：不同 AI 工具可共享同一 MCP 服务器
- **安全**：明确的权限边界和隔离
- **可扩展**：轻松添加新功能，无需修改核心代码

## 2. MCP 客户端实现

**文件**: `src/services/mcp/client.ts:1-600`

```typescript
/**
 * MCPClient 管理所有 MCP 服务器连接
 */
export class MCPClient {
  private connections = new Map<string, MCPServerConnection>()
  private configLoader: ConfigLoader
  private authManager: AuthManager

  constructor(configLoader: ConfigLoader, authManager: AuthManager) {
    this.configLoader = configLoader
    this.authManager = authManager
  }

  /**
   * 初始化所有已配置的 MCP 服务器（延迟连接）
   */
  async initialize(): Promise<void> {
    // 加载配置
    const configs = await this.configLoader.loadMCPConfigs()

    // 注册服务器（不立即连接）
    for (const [id, config] of Object.entries(configs)) {
      this.connections.set(id, {
        id,
        name: config.name || id,
        config,
        client: null as any,
        tools: [],
        resources: [],
        isConnected: false,
        retryCount: 0,
        capabilities: {},
        metadata: {
          createdAt: Date.now()
        }
      })
    }
  }

  /**
   * 获取或创建连接（延迟初始化）
   */
  async ensureConnected(serverId: string): Promise<MCPServerConnection> {
    const connection = this.connections.get(serverId)

    if (!connection) {
      throw new Error(`MCP server not found: ${serverId}`)
    }

    // 如果已连接，直接返回
    if (connection.isConnected) {
      return connection
    }

    // 建立新连接
    return await this.connectServer(connection)
  }

  /**
   * 建立服务器连接
   */
  private async connectServer(
    connection: MCPServerConnection
  ): Promise<MCPServerConnection> {
    try {
      // === 步骤 1: 创建客户端 ===
      const client = new Client({
        name: 'claude-code',
        version: '2.1.88'
      })

      // === 步骤 2: 选择传输层 ===
      const transport = await this.createTransport(connection.config)

      // === 步骤 3: 连接 ===
      await client.connect(transport)

      // === 步骤 4: 初始化（获取能力） ===
      const capabilities = await client.initialize()
      connection.capabilities = capabilities

      // === 步骤 5: 获取工具 ===
      if (capabilities.tools) {
        connection.tools = await this.listTools(client, connection.id)
      }

      // === 步骤 6: 获取资源 ===
      if (capabilities.resources) {
        connection.resources = await this.listResources(client, connection.id)
      }

      // === 步骤 7: 更新状态 ===
      connection.client = client
      connection.isConnected = true
      connection.retryCount = 0
      connection.metadata.lastConnectedAt = Date.now()

      return connection
    } catch (error) {
      // 重试逻辑
      connection.retryCount++
      connection.lastError = String(error)

      if (connection.retryCount >= 3) {
        throw new MCPConnectionError(connection.id, String(error))
      }

      // 指数退避重试
      const delay = Math.pow(2, connection.retryCount) * 1000
      await sleep(delay)

      return await this.connectServer(connection)
    }
  }

  /**
   * 创建传输层
   */
  private async createTransport(config: McpServerConfig): Promise<Transport> {
    switch (config.type) {
      case 'stdio':
        return new StdioClientTransport({
          command: config.command,
          args: config.args,
          env: { ...process.env, ...(config.env || {}) }
        })

      case 'sse':
        return new SSEClientTransport({
          url: config.url,
          headers: await this.getHeaders(config),
          fetch: createFetchWithOAuth(config.oauth)
        })

      case 'http':
        return new StreamableHTTPClientTransport({
          url: config.url,
          headers: await this.getHeaders(config)
        })

      case 'ws':
        return new WebSocketTransport({
          url: config.url,
          headers: await this.getHeaders(config)
        })

      case 'sdk':
        return createSDKTransport(config.name)

      default:
        throw new Error(`Unknown transport: ${(config as any).type}`)
    }
  }

  /**
   * 获取请求头（含认证）
   */
  private async getHeaders(config: McpServerConfig): Promise<Record<string, string>> {
    const headers: Record<string, string> = {}

    if ('headers' in config && config.headers) {
      Object.assign(headers, config.headers)
    }

    // 添加 OAuth 令牌
    if ('oauth' in config && config.oauth) {
      const token = await this.authManager.getToken(config.oauth)
      headers['Authorization'] = `Bearer ${token}`
    }

    return headers
  }

  /**
   * 执行 MCP 工具
   */
  async callTool(
    serverId: string,
    toolName: string,
    input: Record<string, unknown>
  ): Promise<MCPToolResult> {
    const connection = await this.ensureConnected(serverId)

    try {
      const result = await connection.client.request({
        method: 'tools/call',
        params: {
          name: toolName,
          arguments: input
        }
      })

      return result
    } catch (error) {
      if (error instanceof SessionExpiredError) {
        // 清除连接，重试
        connection.isConnected = false
        return await this.callTool(serverId, toolName, input)
      }

      throw error
    }
  }

  /**
   * 读取资源
   */
  async readResource(
    serverId: string,
    uri: string
  ): Promise<string> {
    const connection = await this.ensureConnected(serverId)

    const result = await connection.client.request({
      method: 'resources/read',
      params: { uri }
    })

    return result.content
  }

  /**
   * 列出工具
   */
  private async listTools(
    client: Client,
    serverId: string
  ): Promise<Tool[]> {
    const result = await client.request({
      method: 'tools/list'
    })

    return result.tools || []
  }

  /**
   * 列出资源
   */
  private async listResources(
    client: Client,
    serverId: string
  ): Promise<ServerResource[]> {
    const result = await client.request({
      method: 'resources/list'
    })

    return result.resources || []
  }
}
```

## 3. MCP 工具包装

```typescript
/**
 * 将 MCP 工具包装为 Claude Code 工具
 */
export function createMCPToolWrapper(
  connection: MCPServerConnection,
  mcpTool: Tool
): Tool {
  return {
    name: `mcp:${connection.id}:${mcpTool.name}`,
    description: mcpTool.description,
    inputSchema: mcpTool.inputSchema,

    async invoke(input, context) {
      try {
        // 权限检查
        const permission = await context.canUseTool(
          `mcp:${connection.id}:${mcpTool.name}`,
          input
        )

        if (!permission.allowed) {
          return {
            type: 'error',
            error: `Tool denied: ${permission.reason}`
          }
        }

        // 执行工具
        const mcpClient = new MCPClient(
          new ConfigLoader(),
          new AuthManager()
        )

        const result = await mcpClient.callTool(
          connection.id,
          mcpTool.name,
          input
        )

        return {
          type: 'text',
          text: formatMCPResult(result)
        }
      } catch (error) {
        return {
          type: 'error',
          error: String(error)
        }
      }
    }
  }
}
```

## 4. 配置加载优先级

```
本地项目配置 (.claude/mcp.json)
         ↑ 最高优先级
用户配置 (~/.claude-code/mcp.json)
         ↑
项目设置 (settings.json)
         ↑
企业管理配置（MDM）
         ↑
默认配置
         ↓ 最低优先级
```

## 关键特性

✅ **多种传输方式** - stdio, SSE, HTTP, WebSocket, SDK
✅ **延迟初始化** - 只在需要时连接服务器
✅ **自动重试** - 连接失败时指数退避重试
✅ **OAuth 支持** - 安全的身份认证
✅ **分层配置** - 灵活的配置优先级
✅ **工具包装** - 无缝集成 MCP 工具
