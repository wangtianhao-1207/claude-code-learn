# 3.2 缓存策略与多层缓存设计

## 缓存系统概述

Claude Code 采用多层缓存策略，通过内存缓存、文件缓存和 API 缓存相结合，显著提升性能和降低 API 成本。

---

## 1. 多层缓存架构

```
┌─────────────────────────────────────────────────┐
│          应用缓存请求层                         │
└────────────────┬────────────────────────────────┘
                 │
         ┌───────┴────────┐
         │                │
    ┌────▼──────┐   ┌─────▼──────┐
    │ L1: 内存  │   │ L2: 文件   │
    │ (10-100ms)│   │ (100-1000ms)│
    └────┬──────┘   └─────┬──────┘
         │                │
    ┌────▼──────────┬─────▼──────┐
    │ 命中: 返回    │ 查询未命中 │
    └───────────────┴─────┬──────┘
                          │
                  ┌───────▼──────┐
                  │ L3: 问题 API │
                  │ (1000ms+)    │
                  └───────┬──────┘
                          │
              ┌───────────┴────────┐
              │                    │
         ┌────▼────┐         ┌─────▼────┐
         │ 缓存    │         │ 返回结果 │
         │ L1+L2   │         │(更新缓存)│
         └─────────┘         └──────────┘
```

/**
 * LRU 内存缓存实现
 */
export class LRUCache<K, V> {
  private cache: Map<K, V>
  private maxSize: number
  private hits = 0
  private misses = 0

  constructor(maxSize: number = 100) {
    this.maxSize = maxSize
    this.cache = new Map()
  }

  /**
   * 获取缓存值
   */
  get(key: K): V | undefined {
    if (this.cache.has(key)) {
      const value = this.cache.get(key)!

      // === LRU 逻辑：移到最后 ===
      this.cache.delete(key)
      this.cache.set(key, value)

      this.hits++
      return value
    }

    this.misses++
    return undefined
  }

  /**
   * 设置缓存值
   */
  set(key: K, value: V): void {
    // 如果已存在，删除旧值
    if (this.cache.has(key)) {
      this.cache.delete(key)
    }

    // 添加新值
    this.cache.set(key, value)

    // 超过大小限制时，删除最旧的（最早的）
    if (this.cache.size > this.maxSize) {
      const firstKey = this.cache.keys().next().value
      this.cache.delete(firstKey)
    }
  }

  /**
   * 获取缓存统计
   */
  getStats(): {
    size: number
    hits: number
    misses: number
    hitRate: number
  } {
    const total = this.hits + this.misses
    return {
      size: this.cache.size,
      hits: this.hits,
      misses: this.misses,
      hitRate: total === 0 ? 0 : this.hits / total
    }
  }

  /**
   * 清空缓存
   */
  clear(): void {
    this.cache.clear()
    this.hits = 0
    this.misses = 0
  }
}

/**
 * 文件读取缓存（特化版本）
 */
export class FileReadCache {
  private cache: LRUCache<string, string>
  private checksums = new Map<string, string>()

  constructor(maxFiles: number = 100) {
    this.cache = new LRUCache(maxFiles)
  }

  /**
   * 获取文件内容（带缓存）
   */
  async get(filePath: string): Promise<string | null> {
    // === 1. 尝试从内存缓存获取 ===
    let cached = this.cache.get(filePath)
    if (cached !== undefined) {
      return cached
    }

    // === 2. 从磁盘读取 ===
    try {
      const content = await readFile(filePath, 'utf-8')

      // === 3. 计算文件哈希（用于验证） ===
      const hash = createHash('md5').update(content).digest('hex')
      this.checksums.set(filePath, hash)

      // === 4. 存入缓存 ===
      this.cache.set(filePath, content)

      return content
    } catch (error) {
      return null
    }
  }

  /**
   * 验证缓存有效性
   *
   * 用于文件被外部修改的场景
   */
  async validate(filePath: string): Promise<boolean> {
    try {
      const content = await readFile(filePath, 'utf-8')
      const newHash = createHash('md5').update(content).digest('hex')
      const oldHash = this.checksums.get(filePath)

      if (oldHash !== newHash) {
        // 文件已修改，清除缓存
        this.cache.cache.delete(filePath)
        this.checksums.delete(filePath)
        return false
      }

      return true
    } catch {
      return false
    }
  }

  /**
   * 手动清除单个文件的缓存
   */
  invalidate(filePath: string): void {
    this.cache.cache.delete(filePath)
    this.checksums.delete(filePath)
  }

  /**
   * 获取统计信息
   */
  getStats() {
    return this.cache.getStats()
  }
}

/**
 * 文件级持久化缓存
 */
export class FileCache {
  private cacheDir: string
  private ttl: number  // Time To Live（毫秒）

  constructor(
    cacheDir: string = join(getHomeDir(), '.claude-code', 'cache'),
    ttl: number = 7 * 24 * 60 * 60 * 1000  // 默认 7 天
  ) {
    this.cacheDir = cacheDir
    this.ttl = ttl
    this.ensureDir()
  }

  /**
   * 确保缓存目录存在
   */
  private async ensureDir(): Promise<void> {
    try {
      await mkdir(this.cacheDir, { recursive: true })
    } catch (error) {
      console.error('Failed to create cache directory:', error)
    }
  }

  /**
   * 生成缓存键
   *
   * 使用哈希避免长路径问题
   */
  private getCacheKey(key: string): string {
    return createHash('sha256').update(key).digest('hex').slice(0, 16)
  }

  /**
   * 获取缓存文件路径
   */
  private getCachePath(key: string): string {
    const cacheKey = this.getCacheKey(key)
    return join(this.cacheDir, `${cacheKey}.json`)
  }

  /**
   * 从文件缓存获取
   */
  async get<T>(key: string): Promise<T | null> {
    try {
      const cachePath = this.getCachePath(key)

      // === 1. 检查文件是否存在 ===
      const stat = await fsStat(cachePath)

      // === 2. 检查是否过期 ===
      const age = Date.now() - stat.mtime.getTime()
      if (age > this.ttl) {
        await unlink(cachePath)
        return null
      }

      // === 3. 读取并解析 ===
      const content = await readFile(cachePath, 'utf-8')
      const data = JSON.parse(content)

      return data.value as T
    } catch (error) {
      return null
    }
  }

  /**
   * 写入文件缓存
   */
  async set<T>(key: string, value: T): Promise<void> {
    try {
      const cachePath = this.getCachePath(key)

      const data = {
        key,
        value,
        timestamp: Date.now()
      }

      await writeFile(cachePath, JSON.stringify(data, null, 2))
    } catch (error) {
      console.error('Failed to write cache:', error)
    }
  }

  /**
   * 清理过期缓存
   */
  async cleanup(): Promise<number> {
    let cleaned = 0

    try {
      const files = await readdir(this.cacheDir)

      for (const file of files) {
        const filePath = join(this.cacheDir, file)
        const stat = await fsStat(filePath)

        const age = Date.now() - stat.mtime.getTime()
        if (age > this.ttl) {
          await unlink(filePath)
          cleaned++
        }
      }
    } catch (error) {
      console.error('Failed to cleanup cache:', error)
    }

    return cleaned
  }

  /**
   * 获取缓存大小
   */
  async getSize(): Promise<number> {
    let totalSize = 0

    try {
      const files = await readdir(this.cacheDir)

      for (const file of files) {
        const filePath = join(this.cacheDir, file)
        const stat = await fsStat(filePath)
        totalSize += stat.size
      }
    } catch (error) {
      console.error('Failed to get cache size:', error)
    }

    return totalSize
  }
}

/**
 * 导入必需模块
 */
import { createHash } from 'crypto'
import { mkdir, readdir, unlink } from 'fs/promises'
import { stat as fsStat, promises as fsPromises } from 'fs'

**文件**: `src/services/cache/PromptCache.ts:1-200`

```typescript
/**
 * Claude API 提示缓存（临时缓存）
 *
 * 文档：https://docs.anthropic.com/en/docs/build-a-claude-app/caching
 *
 * 原理：
 * - 发送相同的系统提示和消息前缀时，使用 cache_control 标记
 * - API 会缓存这些内容，后续请求可复用，节省 token 和成本
 * - 缓存命中时，命中部分的 token 成本 = 1/10 的正常成本
 */
export class PromptCache {
  private client: Anthropic
  private cacheStats = {
    hits: 0,
    misses: 0,
    savedTokens: 0
  }

  constructor() {
    this.client = new Anthropic()
  }

  /**
   * 调用 API（启用提示缓存）
   */
  async query(
    systemPrompt: string,
    userMessage: string,
    params: {
      model?: string
      maxTokens?: number
      useCache?: boolean
    } = {}
  ): Promise<APIResponse> {
    const model = params.model ?? 'claude-opus-4-5-20251101'
    const maxTokens = params.maxTokens ?? 2000
    const useCache = params.useCache !== false

    // === 构建消息 ===
    const messages: MessageParam[] = [
      {
        role: 'user',
        content: userMessage
      }
    ]

    // === 构建系统提示（启用缓存） ===
    const system: BlockParam[] = [
      {
        type: 'text',
        text: systemPrompt,
        // 标记为可缓存（缓存 5 分钟）
        cache_control: useCache ? { type: 'ephemeral' } : undefined
      } as any
    ]

    try {
      const response = await this.client.messages.create({
        model,
        max_tokens: maxTokens,
        system,
        messages,
        temperature: 0.5
      } as any)

      // === 提取缓存使用情况 ===
      const usage = (response as any).usage
      if (usage.cache_read_input_tokens > 0) {
        this.cacheStats.hits++
        this.cacheStats.savedTokens +=
          Math.floor(usage.cache_read_input_tokens * 0.9)  // 节省 90%
      } else {
        this.cacheStats.misses++
      }

      return {
        content: response.content[0].type === 'text'
          ? response.content[0].text
          : '',
        usage: {
          inputTokens: usage.input_tokens,
          cacheReadTokens: usage.cache_read_input_tokens ?? 0,
          cacheCreateTokens: usage.cache_creation_input_tokens ?? 0,
          outputTokens: usage.output_tokens
        },
        cacheHit: usage.cache_read_input_tokens > 0
      }
    } catch (error) {
      throw new Error(`API call failed: ${String(error)}`)
    }
  }

  /**
   * 获取缓存统计
   */
  getStats() {
    return {
      ...this.cacheStats,
      hitRate: this.cacheStats.hits +
        this.cacheStats.misses === 0
        ? 0
        : this.cacheStats.hits /
          (this.cacheStats.hits + this.cacheStats.misses)
    }
  }
}

interface APIResponse {
  content: string
  usage: {
    inputTokens: number
    cacheReadTokens: number
    cacheCreateTokens: number
    outputTokens: number
  }
  cacheHit: boolean
}
```

## 5. 缓存策略应用

### 5.1 工具输出缓存

```typescript
/**
 * 工具输出缓存
 *
 * 场景：
 * - 重复调用相同工具（如 ls、cat 同一文件）
 * - 同一工具输出多次使用（如查询 API 结果）
 */
export class ToolOutputCache {
  private memory: LRUCache<string, ToolResult>
  private disk: FileCache

  constructor() {
    this.memory = new LRUCache(50)  // 50 个工具输出
    this.disk = new FileCache()
  }

  /**
   * 生成工具调用的缓存键
   */
  private getCacheKey(tool: Tool, input: Record<string, unknown>): string {
    // 某些工具不需要缓存（如 bash 可能有副作用）
    const noCacheTool = new Set(['bash', 'write', 'edit'])
    if (noCacheTool.has(tool.name)) {
      return ''  // 不缓存
    }

    const key = {
      toolName: tool.name,
      input: JSON.stringify(input)
    }

    return createHash('sha256')
      .update(JSON.stringify(key))
      .digest('hex')
  }

  /**
   * 获取缓存的工具结果
   */
  async get(
    tool: Tool,
    input: Record<string, unknown>
  ): Promise<ToolResult | null> {
    const key = this.getCacheKey(tool, input)
    if (!key) return null

    // === L1: 内存缓存 ===
    let result = this.memory.get(key)
    if (result) return result

    // === L2: 文件缓存 ===
    result = await this.disk.get(key)
    if (result) {
      // 回写到内存缓存
      this.memory.set(key, result)
      return result
    }

    return null
  }

  /**
   * 缓存工具结果
   */
  async set(
    tool: Tool,
    input: Record<string, unknown>,
    result: ToolResult
  ): Promise<void> {
    const key = this.getCacheKey(tool, input)
    if (!key) return

    // === 存入 L1 ===
    this.memory.set(key, result)

    // === 存入 L2（仅保存文本结果） ===
    if (result.type === 'text') {
      await this.disk.set(key, result)
    }
  }
}
```

 * 利用 Claude API 的提示缓存来加速对话

export class MessageHistoryCache {
  private promptCache: PromptCache
  private lastCachedMessage = 0

  constructor() {
    this.promptCache = new PromptCache()
  }

  /**
   * 获取可缓存的消息前缀
   *
   * 策略：缓存最近的 N 条消息的前 M 个
   * - 这些消息通常会在后续请求中重复使用
   * - API 会自动缓存这些内容
   */
  getCacheableMessages(
    messages: Message[],
    upTo: number = 5
  ): Message[] {
    // 只缓存前 upTo 条消息
    // 这保证了缓存的稳定性（新消息不会影响旧缓存）
    return messages.slice(0, Math.min(upTo, messages.length))
  }
}

## 7. 缓存成本效益

### 计算示例

```
场景：编写一个 1000 行的代码生成器
- 系统提示: 2000 token
- 代码库上下文: 5000 token
- 用户消息: 100 token
- 总输入: 7100 token

无缓存成本：
- 每次调用: 7100 * $0.003 = $0.0213

启用提示缓存后：
- 首次调用: 2000 * $0.003 + 5100 * $0.0003 = $6.30 + $1.53 = $7.83
- 后续调用: 7100 * $0.0003 = $2.13
- 缓存节省: 90% 的成本用于缓存内容

10 次调用对比：
- 无缓存: $0.213
- 有缓存: $7.83 + (9 * $2.13) = $26.96
- 实际节省: 不在 token，而在于缓存命中率

关键：缓存对长期、重复工作流更有价值
```

## 关键特性

✅ **多层缓存** - L1 内存 → L2 文件 → L3 API
✅ **LRU 算法** - 自动淘汰最少使用的缓存项
✅ **提示缓存** - 利用 Claude API 的原生缓存能力
✅ **自动过期** - 支持 TTL，自动清理过期数据
✅ **统计监控** - 详细的缓存命中率和成本分析
✅ **智能清理** - 定期清理过期缓存，防止磁盘爆满
✅ **写保护** - 敏感操作（bash、edit）禁用缓存
