# 3.1 上下文压缩机制 - 深度剖析

## 概述

Claude Code 实现了多层次的上下文压缩策略，在有限的 Token 预算内最大化信息密度：

- **消息级压缩**：汇总旧消息
- **工具输出级压缩**：截断过长结果
- **文件级压缩**：分片读取大文件
- **提示缓存**：复用稳定的系统提示

### 压缩触发条件
- Token 使用超过 75% → 警告
- Token 使用超过 85% → 自动压缩
- Token 使用超过 95% → 拒绝新请求

/**
 * Token 预算追踪和管理
 */

export class TokenBudget {
  private readonly maxTokens: number
  private inputTokens = 0
  private outputTokens = 0
  private cacheCreationTokens = 0
  private cacheReadTokens = 0
  private readonly model: string

  // 门槛值
  private readonly warningThreshold: number  // 75%
  private readonly criticalThreshold: number // 85%
  private readonly absoluteMax: number      // 95%

  constructor(maxTokens: number, model: string) {
    this.maxTokens = maxTokens
    this.model = model
    this.warningThreshold = Math.floor(maxTokens * 0.75)
    this.criticalThreshold = Math.floor(maxTokens * 0.85)
    this.absoluteMax = Math.floor(maxTokens * 0.95)
  }

  /**
   * 报告 Token 使用量
   */
  reportUsage(usage: {
    inputTokens: number
    outputTokens: number
    cacheCreationTokens?: number
    cacheReadTokens?: number
  }): TokenBudgetStatus {
    this.inputTokens += usage.inputTokens
    this.outputTokens += usage.outputTokens
    this.cacheCreationTokens += usage.cacheCreationTokens || 0
    this.cacheReadTokens += usage.cacheReadTokens || 0

    return this.getStatus()
  }

  /**
   * 获取当前状态
   */
  getStatus(): TokenBudgetStatus {
    const total = this.inputTokens + this.outputTokens
    const ratio = total / this.maxTokens

    return {
      total,
      inputTokens: this.inputTokens,
      outputTokens: this.outputTokens,
      cacheTokens: {
        creation: this.cacheCreationTokens,
        read: this.cacheReadTokens
      },
      remaining: this.maxTokens - total,
      ratio,
      status: this.calculateStatus(ratio),
      canContinue: total < this.absoluteMax,
      shouldCompress: total > this.criticalThreshold,
      model: this.model
    }
  }

  /**
   * 检查是否可继续
   */
  canAcceptNewMessage(estimatedTokens: number): boolean {
    const total = this.inputTokens + this.outputTokens
    return total + estimatedTokens < this.maxTokens
  }

  /**
   * 估算消息的 Token 数
   */
  estimateTokens(text: string): number {
    // Claude 使用的分词器
    // 粗略估算：1 token ≈ 4 字符
    return Math.ceil(text.length / 4)
  }

  private calculateStatus(ratio: number): 'normal' | 'warning' | 'critical' | 'exceeded' {
    if (ratio >= 1.0) return 'exceeded'
    if (ratio >= this.absoluteMax / this.maxTokens) return 'critical'
    if (ratio >= this.criticalThreshold / this.maxTokens) return 'warning'
    return 'normal'
  }
}

export type TokenBudgetStatus = {
  total: number
  inputTokens: number
  outputTokens: number
  cacheTokens: {
    creation: number
    read: number
  }
  remaining: number
  ratio: number
  status: 'normal' | 'warning' | 'critical' | 'exceeded'
  canContinue: boolean
  shouldCompress: boolean
  model: string
}

// src/services/compact/messageCompression.ts:80-250
/**
 * 消息历史压缩算法
 * 保留：初始消息 + 最近消息，汇总中间消息
 */

export function compressMessages(
  messages: Message[],
  targetTokenRatio: number = 0.7
): Message[] {
  if (messages.length <= 3) {
    return messages // 太短，不压缩
  }

  // === 第1步：计算当前 Token 数 ===
  const originalTokenCount = countMessageTokens(messages)
  const targetTokenCount = Math.floor(originalTokenCount * targetTokenRatio)

  // === 第2步：识别保留和压缩的部分 ===

  // 保留：初始消息（通常是系统消息）
  const kept: Message[] = [messages[0]]

  // 保留：最近的 N 条消息（定义为最后 2-4 条，取决于大小）
  const recentMessages: Message[] = []
  let recentTokens = 0
  const recentTokenBudget = Math.floor(targetTokenCount * 0.3)

  for (let i = messages.length - 1; i > 0; i--) {
    const msg = messages[i]
    const tokens = countMessageTokens([msg])

    if (recentTokens + tokens > recentTokenBudget) {
      break
    }

    recentMessages.unshift(msg)
    recentTokens += tokens
  }

  // 需要压缩的中间消息
  const middleStart = 1
  const middleEnd = messages.length - recentMessages.length
  const middleMessages = messages.slice(middleStart, middleEnd)

  // === 第3步：创建汇总消息 ===
  let summary: AssistantMessage | null = null

  if (middleMessages.length > 0) {
    summary = await createMessageSummary(
      middleMessages,
      Math.floor(targetTokenCount * 0.7)
    )

    kept.push(summary)
  }

  // === 第4步：添加最近消息 ===
  kept.push(...recentMessages)

  // === 第5步：验证压缩效果 ===
  const compressedTokenCount = countMessageTokens(kept)
  const compressionRatio = compressedTokenCount / originalTokenCount

  console.log(
    `Compressed messages: ` +
    `${messages.length} → ${kept.length} messages, ` +
    `${originalTokenCount} → ${compressedTokenCount} tokens ` +
    `(${(compressionRatio * 100).toFixed(1)}%)`
  )

  return kept
}

/**
 * 创建消息汇总
 */
async function createMessageSummary(
  messages: Message[],
  tokenBudget: number
): Promise<AssistantMessage> {
  // 提取文本内容
  const content = messages
    .map(msg => {
      if (msg.type === 'user') {
        return `User: ${msg.content}`
      } else if (msg.type === 'assistant') {
        return `Assistant: ${extractTextContent(msg.content)}`
      } else if (msg.type === 'tool_result') {
        return `Tool (${msg.toolName}): ${
          typeof msg.content === 'string'
            ? msg.content
            : extractTextContent([msg.content])
        }`
      }
      return ''
    })
    .filter(c => c)
    .join('\n\n')

  // 如果内容过长，截断
  const truncated = truncateText(content, tokenBudget * 4)

  return {
    type: 'assistant',
    uuid: generateUUID(),
    role: 'assistant',
    content: [{
      type: 'text',
      text: `[Previous conversation summary from ${messages.length} messages]\n\n${truncated}`
    }],
    timestamp: Date.now(),
    stopReason: 'summary'
  }
}

/**
 * 计算消息组的总 Token 数
 */
function countMessageTokens(messages: Message[]): number {
  return messages.reduce((sum, msg) => {
    let text = ''

    if (msg.type === 'user') {
      text = msg.content
    } else if (msg.type === 'assistant') {
      text = extractTextContent(msg.content)
    } else if (msg.type === 'tool_result') {
      text = typeof msg.content === 'string'
        ? msg.content
        : extractTextContent([msg.content])
    }

    // 粗略估算：1 token ≈ 4 字符
    return sum + Math.ceil(text.length / 4)
  }, 0)
}

/**
 * 截断文本到指定长度
 */
function truncateText(text: string, maxLength: number): string {
  if (text.length <= maxLength) {
    return text
  }

  // 优先保留开头和结尾
  const half = Math.floor(maxLength / 2)
  return (
    text.substring(0, half) +
    '\n\n... truncated ...\n\n' +
    text.substring(text.length - half)
  )
}

/**
 * 提取内容块中的文本
 */
function extractTextContent(blocks: ContentBlock[]): string {
  return blocks
    .filter(b => b.type === 'text')
    .map(b => (b as TextBlock).text)
    .join('\n')
}

// src/services/compact/outputTruncation.ts:50-200
const MAX_TOOL_OUTPUT_SIZE = 10_000 // 10KB
const MAX_TOOL_OUTPUT_LINES = 500

/**
 * 截断工具输出
 */
export function trimToolOutput(
  output: string,
  maxSize: number = MAX_TOOL_OUTPUT_SIZE,
  maxLines: number = MAX_TOOL_OUTPUT_LINES
): string {
  // === 检查 1：行数限制 ===
  const lines = output.split('\n')
  let trimmedLines = lines

  if (lines.length > maxLines) {
    // 保留前后各 25% 的行数，中间削减
    const quarter = Math.floor(maxLines / 4)
    trimmedLines = [
      ...lines.slice(0, quarter),
      `... [${lines.length - maxLines} lines omitted] ...`,
      ...lines.slice(-quarter)
    ]
  }

  let result = trimmedLines.join('\n')

  // === 检查 2：大小限制 ===
  if (result.length > maxSize) {
    const half = Math.floor(maxSize / 2)
    result = (
      result.substring(0, half) +
      `\n\n... [${result.length - maxSize} bytes omitted] ...\n\n` +
      result.substring(result.length - half)
    )
  }

  return result
}

/**
 * 智能截断：保留最有用的部分
 */
export function smartTrimToolOutput(
  output: string,
  context: TrimContext
): string {
  // 识别关键部分
  const sections = identifySections(output)

  // 优先级排序
  const prioritized = sections.sort((a, b) => {
    // 错误信息优先级最高
    if (a.type === 'error') return -1
    if (b.type === 'error') return 1

    // 其次是摘要/标题
    if (a.type === 'summary') return -1
    if (b.type === 'summary') return 1

    // 最后是正常内容，按出现顺序
    return a.position - b.position
  })

  // 累积直到达到大小限制
  let accumulated = ''
  for (const section of prioritized) {
    const newSize = accumulated.length + section.content.length

    if (newSize > MAX_TOOL_OUTPUT_SIZE) {
      // 添加省略号
      accumulated += '\n\n... output truncated ...\n'
      break
    }

    accumulated += section.content
  }

  return accumulated
}

interface TrimContext {
  toolName: string
  isError: boolean
  outputType?: 'json' | 'text' | 'mixed'
}

interface Section {
  type: 'error' | 'summary' | 'content' | 'metadata'
  content: string
  position: number
}

/**
 * 识别输出中的关键部分
 */
function identifySections(output: string): Section[] {
  const sections: Section[] = []

  // 识别错误块（通常以 Error: 或 Exception: 开头）
  const errorMatches = [...output.matchAll(/^(Error|Exception|Failed):.+$/gm)]
  for (const match of errorMatches) {
    sections.push({
      type: 'error',
      content: match[0],
      position: match.index || 0
    })
  }

  // 识别摘要块（通常在开头）
  if (output.length < 1000) {
    sections.push({
      type: 'summary',
      content: output.substring(0, 200),
      position: 0
    })
  }

  // 如果没有发现任何特殊部分，返回整个内容
  if (sections.length === 0) {
    sections.push({
      type: 'content',
      content: output,
      position: 0
    })
  }

  return sections
}

// src/tools/FileReadTool/largeFileHandling.ts:80-200
const MAX_FILE_SIZE_DIRECT = 1_000_000 // 1MB 直接读取
const CHUNK_SIZE = 50_000 // 50KB 分片大小

/**
 * 智能文件读取
 */
export async function readFileIntelligent(
  filePath: string,
  context: {
    offset?: number
    limit?: number
    encoding?: BufferEncoding
  } = {}
): Promise<FileContent> {
  const { offset = 0, limit, encoding = 'utf-8' } = context

  // === 步骤 1：获取文件信息 ===
  const stat = await fstat(filePath)

  if (!stat.isFile()) {
    throw new Error(`Not a file: ${filePath}`)
  }

  // === 步骤 2：根据大小选择策略 ===

  if (stat.size <= MAX_FILE_SIZE_DIRECT) {
    // 小文件：直接读取
    return await readFileDirectly(filePath, offset, limit, encoding)
  } else {
    // 大文件：分片读取
    return await readFileinChunks(filePath, offset, limit, encoding)
  }
}

/**
 * 直接读取文件
 */
async function readFileDirectly(
  filePath: string,
  offset: number,
  limit: number | undefined,
  encoding: BufferEncoding
): Promise<FileContent> {
  let content = await readFile(filePath, encoding)

  // 应用偏移和限制
  if (offset > 0 || limit) {
    const end = limit ? offset + limit : undefined
    content = content.slice(offset, end)
  }

  return {
    content,
    size: content.length,
    isPartial: offset > 0 || limit !== undefined
  }
}

/**
 * 分片读取大文件
 */
async function readFileinChunks(
  filePath: string,
  offset: number,
  limit: number | undefined,
  encoding: BufferEncoding
): Promise<FileContent> {
  const chunks: string[] = []
  let bytesRead = 0
  const targetBytes = limit || Infinity

  const fd = await open(filePath, 'r')

  try {
    let position = offset

    while (bytesRead < targetBytes) {
      const chunkToRead = Math.min(
        CHUNK_SIZE,
        targetBytes - bytesRead
      )

      const { bytesRead: read, buffer } = await fd.read(
        Buffer.alloc(chunkToRead),
        0,
        chunkToRead,
        position
      )

      if (read === 0) break

      chunks.push(buffer.toString(encoding, 0, read))
      bytesRead += read
      position += read
    }
  } finally {
    await fd.close()
  }

  return {
    content: chunks.join(''),
    size: bytesRead,
    isPartial: true
  }
}

interface FileContent {
  content: string
  size: number
  isPartial: boolean
}

export interface CacheControlEphemeral {
  type: 'ephemeral'
}

/**
 * 使用提示缓存优化 API 调用
 */
export async function buildPromptWithCaching(
  systemPrompt: string,
  messages: Message[]
): Promise<{
  system: (string | { type: 'text'; text: string; cache_control: CacheControlEphemeral })[]
  messages: MessageParam[]
}> {
  // 系统提示可以被缓存（在整个会话中保持稳定）
  const cachedSystem = {
    type: 'text' as const,
    text: systemPrompt,
    cache_control: { type: 'ephemeral' as const }
  }

  // 消息列表（最后一条消息不缓存，因为即将添加新内容）
  const messagesToCache = messages.slice(0, -1)
  const lastMessage = messages[messages.length - 1]

  // 构建消息数组
  const systemArray = [cachedSystem]

  const messageParams = [
    ...messagesToCache.map(msg => ({
      role: msg.type === 'assistant' ? 'assistant' : 'user',
      content: formatMessageContent(msg.content),
      cache_control: {
        type: 'ephemeral' as const
      }
    })),
    // 最后一条消息不缓存
    {
      role: lastMessage.type === 'assistant' ? 'assistant' : 'user',
      content: formatMessageContent(lastMessage.content)
    }
  ]

  return {
    system: systemArray,
    messages: messageParams
  }
}

/**
 * 分析缓存效果
 */
export function analyzePromptCacheUsage(response: MessageCreateParamsNonStreaming): CacheAnalysis {
  const usage = response.usage

  return {
    inputTokens: usage.input_tokens,
    cacheCreationTokens: usage.cache_creation_input_tokens || 0,
    cacheReadTokens: usage.cache_read_input_tokens || 0,
    outputTokens: usage.output_tokens,
    totalTokens: usage.input_tokens + (usage.output_tokens || 0),
    effectiveCacheRate: usage.cache_read_input_tokens
      ? usage.cache_read_input_tokens / (usage.input_tokens + usage.cache_read_input_tokens)
      : 0,
    efficiency: {
      withoutCache: usage.input_tokens + usage.cache_creation_input_tokens,
      withCache: usage.cache_read_input_tokens,
      saved: (usage.cache_read_input_tokens || 0) - Math.ceil((usage.cache_read_input_tokens || 0) * 0.1)
    }
  }
}

interface CacheAnalysis {
  inputTokens: number
  cacheCreationTokens: number
  cacheReadTokens: number
  outputTokens: number
  totalTokens: number
  effectiveCacheRate: number
  efficiency: {
    withoutCache: number
    withCache: number
    saved: number

/**
 * 监控 Token 使用，自动触发压缩
 */

export class AutoCompressionManager {
  private tokenBudget: TokenBudget
  private hasCompressed = false
  private compressionHistory: CompressionEvent[] = []

  constructor(tokenBudget: TokenBudget) {
    this.tokenBudget = tokenBudget
  }

  /**
   * 检查是否需要压缩
   */
  checkAndCompress(
    messages: Message[]
  ): { messages: Message[]; compressed: boolean } {
    const status = this.tokenBudget.getStatus()

    // 首次警告
    if (status.status === 'warning' && !this.hasCompressed) {
      console.warn(
        `⚠️ Token usage at ${(status.ratio * 100).toFixed(1)}% ` +
        `(${status.total}/${status.remaining + status.total}). ` +
        `Consider wrapping up.`
      )
    }

    // 自动压缩条件
    if (status.shouldCompress && !this.hasCompressed) {
      console.log(`🔄 Auto-compressing message history...`)

      const compressed = compressMessages(messages, 0.65)
      this.hasCompressed = true

      this.compressionHistory.push({
        timestamp: Date.now(),
        originalMessages: messages.length,
        compressedMessages: compressed.length,
        tokensSaved: countMessageTokens(messages) - countMessageTokens(compressed)
      })

      return {
        messages: compressed,
        compressed: true
      }
    }

    return {
      messages,
      compressed: false
    }
  }

  /**
   * 获取压缩历史
   */
  getCompressionHistory(): CompressionEvent[] {
    return [...this.compressionHistory]
  }
}

interface CompressionEvent {
  timestamp: number
  originalMessages: number
  compressedMessages: number
  tokensSaved: number
}
