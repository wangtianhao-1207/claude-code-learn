```typescript
/**
 * Claude Code CLI 主入口
 */
async function main() {
  try {
    // === 阶段 1: 环境初始化（~50ms） ===
    console.log('🚀 Initializing Claude Code...')

    const env = initializeEnvironment()
    const homeDir = getHomeDir()
    const configDir = join(homeDir, '.claude-code')

    // 确保必要的目录存在
    await ensureDirectoriesExist([
      configDir,
      join(configDir, 'sessions'),
      join(configDir, 'memory'),
      join(configDir, 'agents'),
      join(configDir, 'logs')
    ])

    // === 阶段 2: 并行加载关键资源（~200ms） ===
    console.log('📦 Loading configuration...')

    const [
      settings,
      mcpConfigs,
      policies,
      agents
    ] = await Promise.all([
      loadUserSettings(configDir),           // 用户设置
      loadMCPConfigs(configDir),             // MCP 配置
      loadPolicies(),                        // 安全策略
      listAvailableAgents()                  // 代理定义
    ])

    // === 阶段 3: 创建应用状态（~50ms） ===
    const appState: AppState = {
      sessionId: generateSessionId(),
      isRemoteSession: env.isRemote,
      messages: [],
      isWaitingForResponse: false,
      tools: await loadTools(),              // 加载工具
      skills: await loadSkills(),            // 加载技能
      settings,
      agents,
      cwd: process.cwd(),
      projectRoot: await findProjectRoot(process.cwd()),
      tokenUsage: {
        inputTokens: 0,
        outputTokens: 0,
        totalTokens: 0,
        limit: settings.tokenLimit || 100_000,
        ratio: 0,
        status: 'normal'
      },
      costTracking: {
        totalCost: 0,
        costByModel: {},
        costByTool: {}
      },
      diagnostics: {}
    }

    // === 阶段 4: 初始化状态存储（~50ms） ===
    console.log('💾 Setting up session state...')

    const store = createAppStateStore(
      appState,
      onChangeAppState
    )

    // === 阶段 5: 尝试加载前一个会话（~100ms） ===
    if (settings.resumeLastSession) {
      const lastSessionId = await getLastSessionId()
      if (lastSessionId) {
        const restored = await loadSession(lastSessionId)
        if (restored) {
          console.log(`📂 Restored session: ${lastSessionId}`)
          store.setState(restored.appState)
        }
      }
    }

    // === 阶段 6: 初始化 MCP 连接（延迟）（~0ms） ===
    // MCP 连接在首次使用时才建立（延迟初始化）

    // === 阶段 7: 启动 UI（React） ===
    console.log('🎨 Starting UI...')

    const app = await startUI(store, {
      mode: env.isInteractive ? 'interactive' : 'batch',
      remoteSession: env.isRemote,
      debugMode: settings.debugMode
    })

    // === 阶段 8: 进入主循环 ===
    await runREPL(app)

    // === 阶段 9: 优雅关闭 ===
    await shutdown(app, store)

  } catch (error) {
    console.error('❌ Fatal error:', error)
    process.exit(1)
  }
}

/**
 * 环境初始化
 */
function initializeEnvironment() {
  return {
    isInteractive: process.stdin.isTTY,
    isRemote: process.env.CLAUDE_CODE_REMOTE === '1',
    debugMode: process.env.DEBUG === '1',
    apiKey: process.env.ANTHROPIC_API_KEY,
    homeDir: expandHome('~')
  }
}

/**
 * 加载用户设置
 */
async function loadUserSettings(configDir: string): Promise<UserSettings> {
  const settingsPath = join(configDir, 'settings.json')

  try {
    const content = await readFile(settingsPath, 'utf-8')
    return JSON.parse(content)
  } catch {
    // 文件不存在或无效，返回默认值
    return {
      permissionMode: 'default',
      theme: 'auto',
      tokenLimit: 100_000,
      autoSave: true,
      resumeLastSession: true,
      debugMode: false
    }
  }
}

/**
 * 查找项目根目录
 */
async function findProjectRoot(cwd: string): Promise<string> {
  let current = cwd

  while (current !== '/') {
    // 检查标志文件
    const hasGit = await pathExists(join(current, '.git'))
    const hasPackageJson = await pathExists(join(current, 'package.json'))
    const hasClaudeMd = await pathExists(join(current, 'claude.md'))

    if (hasGit || hasPackageJson || hasClaudeMd) {
      return current
    }

    current = path.dirname(current)
  }

  // 默认返回当前工作目录
  return cwd
}
```

```typescript
// src/tools.ts:1-100

/**
 * 加载所有工具
 */
export async function loadTools(): Promise<Tool[]> {
  const tools: Tool[] = [
    // 内置工具（同步加载）
    ...Object.values(BUILTIN_TOOLS),

    // 插件工具（异步加载）
    ...(await loadPluginTools())
  ]

  // 验证工具完整性
  for (const tool of tools) {
    validateToolDefinition(tool)
  }

  return tools
}

/**
 * 验证工具定义
 */
function validateToolDefinition(tool: Tool): void {
  if (!tool.name || !tool.description || !tool.invoke) {
    throw new Error(`Invalid tool definition: ${tool.name}`)
  }

  if (!tool.inputSchema) {
    console.warn(`Tool ${tool.name} missing inputSchema`)
  }
}
```

```typescript
/**
 * Read-Eval-Print Loop
 */
async function runREPL(app: App): Promise<void> {
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
  })

  const prompt = () => {
    rl.question('> ', async (input) => {
      try {
        if (input.trim() === '') {
          prompt()
          return
        }

        // === 处理斜杠命令 ===
        const command = parseCommand(input)
        if (command) {
          // 执行命令
          await executeCommand(command.name, command.args, {
            appState: app.store.getState(),
            setAppState: app.store.setState,
            tools: app.store.getState().tools,
            commands: BUILTIN_COMMANDS,
            canUseTool: app.checkToolPermission,
            abortController: new AbortController(),
            executeCommand: (name, args) => executeCommand(name, args, {
              appState: app.store.getState(),
              setAppState: app.store.setState,
              tools: app.store.getState().tools,
              commands: BUILTIN_COMMANDS,
              canUseTool: app.checkToolPermission,
              abortController: new AbortController(),
              executeCommand: () => Promise.resolve('')
            })
          })
        } else {
          // 作为用户消息处理
          await app.processUserInput(input)
        }

        // 继续循环
        prompt()
      } catch (error) {
        console.error('Error:', error)
        prompt()
      }
    })
  }

  // 启动循环
  prompt()

  // 优雅关闭处理
  process.on('SIGINT', async () => {
    console.log('\n\nShutting down...')
    rl.close()
    await app.shutdown()
    process.exit(0)
  })
}
```

```typescript
/**
 * 应用关闭时的清理
 */
async function shutdown(app: App, store: AppStateStore): Promise<void> {
  console.log('🛑 Shutting down...')

  // === 步骤 1: 保存当前会话 ===
  const appState = store.getState()
  const sessionPath = SESSION_STORAGE_PATHS.sessionDir(appState.sessionId)

  const snapshot = {
    metadata: {
      id: appState.sessionId,
      status: 'closed',
      lastModified: Date.now(),
      messageCount: appState.messages.length
    },
    messages: appState.messages
  }

  await writeFile(
    join(sessionPath, 'snapshot.json'),
    JSON.stringify(snapshot, null, 2)
  )

  // === 步骤 2: 关闭 MCP 连接 ===
  for (const [, connection] of appState.mcpConnections) {
    try {
      await connection.client.disconnect()
    } catch {
      // 忽略错误
    }
  }

  // === 步骤 3: 清理临时文件 ===
  // TODO: 实现临时文件清理

  // === 步骤 4: 记录关闭事件 ===
  console.log('👋 Goodbye!')
}
```

## 启动时间目标

- **总启动时间**: < 1 秒
- **环境初始化**: < 50ms
- **配置加载**: < 100ms
- **应用状态创建**: < 50ms
- **会话恢复**: < 200ms（如果有）
- **UI 启动**: < 200ms
