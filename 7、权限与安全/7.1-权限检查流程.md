# 7.1 权限检查流程 - 深度剖析

## 五层权限检查架构

Claude Code 实现了一个分层的权限检查系统，确保安全同时保持灵活性：

```
Tool Invocation
     ↓
[Layer 1: Global Permission Mode]
  - 'deny': 拒绝所有
  - 'ask': 每次询问
  - 'default': 默认允许（推荐）
  - 'auto': 自动批准
     ↓
[Layer 2: Tool-Level Deny Rules]
  - 检查工具是否在黑名单中
  - 例如: deny_tools: ['bash', 'write']
     ↓
[Layer 3: Tool-Level Allow Rules]
  - 如果指定允许列表，只允许列表中的工具
  - 例如: allowed_tools: ['read', 'glob']
     ↓
[Layer 4: Operation-Specific Checks]
  - 工具特定的权限逻辑
  - 例如：Bash 的危险命令检查
  - 例如：File Write 的目录检查
     ↓
[Layer 5: User Interactive Approval]
  - 如果前面的检查需要用户确认
  - 显示 UI 提示，等待用户输入
     ↓
Execution or Rejection
```

```typescript
export type PermissionMode = 'deny' | 'ask' | 'default' | 'auto'

export function getPermissionMode(settings: UserSettings): PermissionMode {
  // 优先级：显式设置 > 环境变量 > 默认值
  if (process.env.CLAUDE_CODE_PERMISSION_MODE) {
    return process.env.CLAUDE_CODE_PERMISSION_MODE as PermissionMode
  }

  return settings.permissionMode || 'default'
}

/**
 * 处理全局权限模式
 */
async function checkGlobalPermissionMode(
  toolName: string,
  mode: PermissionMode
): Promise<PermissionCheckResult> {
  switch (mode) {
    case 'deny':
      return {
        allowed: false,
        reason: 'global_deny_mode'
      }

    case 'ask':
      return {
        allowed: null,
        reason: 'requires_approval',
        requiresApproval: true
      }

    case 'auto':
      return {
        allowed: true,
        reason: 'auto_approved'
      }

    case 'default':
    default:
      // 继续下一层检查
      return {
        allowed: null,
        reason: 'continue'
      }
  }
}
```

```typescript
// === Bash 命令检查 ===

export async function checkBashOperation(
  command: string,
  context: PermissionContext
): Promise<PermissionCheckResult> {
  const rules = TOOL_RULES.bash?.operationRules?.bash

  if (!rules) {
    return { allowed: true, reason: 'no_rules' }
  }

  // === 检查 1: 禁止的模式 ===
  for (const pattern of rules.forbiddenPatterns) {
    if (pattern.test(command)) {
      return {
        allowed: false,
        reason: 'forbidden_pattern',
        details: `Command matches forbidden pattern: ${pattern.source}`
      }
    }
  }

  // === 检查 2: 允许的前缀 ===
  if (rules.allowedPrefixes && rules.allowedPrefixes.length > 0) {
    const isAllowed = rules.allowedPrefixes.some(prefix =>
      command.startsWith(prefix)
    )
    if (!isAllowed) {
      return {
        allowed: false,
        reason: 'not_in_allowed_list'
      }
    }
  }

  // === 检查 3: 用户自定义规则 ===
  const customRules = context.appState.settings.bashRules || []
  for (const rule of customRules) {
    const regex = new RegExp(rule.pattern)
    if (rule.type === 'deny' && regex.test(command)) {
      return {
        allowed: false,
        reason: 'custom_rule',
        details: `Command matches custom deny rule: ${rule.name}`
      }
    }
  }

  return { allowed: true, reason: 'bash_check_passed' }
}

// === 文件写入检查 ===

export async function checkFileWriteOperation(
  filePath: string,
  context: PermissionContext
): Promise<PermissionCheckResult> {
  const rules = TOOL_RULES.write?.operationRules?.fileWrite

  if (!rules) {
    return { allowed: true, reason: 'no_rules' }
  }

  // === 检查 1: 系统目录保护 ===
  const normalized = path.normalize(path.resolve(filePath))

  for (const forbiddenDir of rules.forbiddenDirs) {
    if (normalized.startsWith(forbiddenDir)) {
      return {
        allowed: false,
        reason: 'system_directory_protected',
        details: `Cannot write to ${forbiddenDir}`
      }
    }
  }

  // === 检查 2: 基础路径限制 ===
  if (rules.allowedBasePath) {
    if (!normalized.startsWith(rules.allowedBasePath)) {
      return {
        allowed: false,
        reason: 'outside_allowed_path'
      }
    }
  }

  // === 检查 3: 敏感文件保护 ===
  const basename = path.basename(normalized)
  if (rules.forbiddenFiles.includes(basename)) {
    return {
      allowed: null,
      reason: 'requires_approval',
      requiresApproval: true,
      promptMessage: `About to write to sensitive file: ${basename}`
    }
  }

  return { allowed: true, reason: 'file_write_check_passed' }
}

// === 网络访问检查 ===

export async function checkNetworkOperation(
  url: string,
  context: PermissionContext
): Promise<PermissionCheckResult> {
  const rules = TOOL_RULES.web_search?.operationRules?.network

  if (!rules) {
    return { allowed: true, reason: 'no_rules' }
  }

  // === 检查 1: 本地网络 ===
  if (!rules.allowLocalhost) {
    const urlObj = new URL(url)
    if (['localhost', '127.0.0.1', '::1'].includes(urlObj.hostname)) {
      return {
        allowed: false,
        reason: 'localhost_forbidden'
      }
    }
  }

  // === 检查 2: 域名白名单 ===
  if (rules.allowedDomains && rules.allowedDomains.length > 0) {
    const urlObj = new URL(url)
    const isAllowed = rules.allowedDomains.some(domain =>
      urlObj.hostname.endsWith(domain)
    )
    if (!isAllowed) {
      return {
        allowed: false,
        reason: 'domain_not_allowed'
      }
    }
  }

  // === 检查 3: 域名黑名单 ===
  if (rules.forbiddenDomains) {
    const urlObj = new URL(url)
    const isForbidden = rules.forbiddenDomains.some(domain =>
      urlObj.hostname.endsWith(domain)
    )
    if (isForbidden) {
      return {
        allowed: false,
        reason: 'domain_forbidden'
      }
    }
  }

  return { allowed: true, reason: 'network_check_passed' }
}
```

```typescript
export function PermissionPrompt({
  toolName,
  operation,
  reason,
  onApprove,
  onDeny
}: {
  toolName: string
  operation: string
  reason: string
  onApprove: () => void
  onDeny: () => void
}) {
  return (
    <Box borderStyle="round" borderColor="yellow" padding={1}>
      <Box marginBottom={1}>
        <Text color="yellow" bold>
          ⚠️ Permission Required
        </Text>
      </Box>

      <Box marginBottom={1}>
        <Text>Tool: <Text color="cyan">{toolName}</Text></Text>
        <Text>Operation: {operation}</Text>
        <Text dimColor>{reason}</Text>
      </Box>

      <Box>
        <Box marginRight={2}>
          <SelectInput
            items={[
              { label: 'Approve', value: 'approve' },
              { label: 'Deny', value: 'deny' }
            ]}
            onSelect={(item) => {
              if (item.value === 'approve') {
                onApprove()
              } else {
                onDeny()
              }
            }}
          />
        </Box>
      </Box>
    </Box>
  )
}
```

```typescript
// src/utils/permissions/auditLog.ts

export class PermissionAuditLog {
  private entries: PermissionAuditEntry[] = []

  log(entry: PermissionAuditEntry) {
    this.entries.push({
      ...entry,
      timestamp: Date.now()
    })

    // 持久化
    this.persist()
  }

  /**
   * 获取被拒绝的操作
   */
  getDeniedOperations(limit = 100) {
    return this.entries
      .filter(e => e.decision === 'denied')
      .slice(-limit)
  }

  /**
   * 获取特定工具的访问历史
   */
  getToolHistory(toolName: string, limit = 100) {
    return this.entries
      .filter(e => e.toolName === toolName)
      .slice(-limit)
  }

  private async persist() {
    const logPath = join(getHomeDir(), '.claude-code', 'permission-audit.log')
    const lines = this.entries
      .map(e => JSON.stringify(e))
      .join('\n')

    await appendFile(logPath, lines + '\n')
  }
}
```
